560static TEnumTypePacket prvFindIncomingPacket (uint8_t *const pucPacketInWhole)
561{
562 uint16_t usPos = 0;
563 uint16_t usKolByteIn; // The number of bytes read from the stream buffer
564 TEnumTypePacket eTypePacketIn = TYPE_PACKET_NOT_FOUND;
565 uint8_t pucPacketIn[MAX_SIZE_PACKET];
566
567 assert_param(pucPacketInWhole != NULL);
568
569 if (pucPacketInWhole == NULL)
570 goto return_prvFindIncomingPacket;
571
572 usKolByteIn = xStreamBufferReceive(xStreamBufferReceiveToTaskInterface, (void*)pucPacketIn, sizeof(pucPacketIn), 0);
573
574 // Find preambule
575 for ( ; usPos < (usKolByteIn - 2); usPos++)
576 {
577 if ((pucPacketIn[usPos] == (PREAMBULE_PACKET & 0xff00)>>8) && (pucPacketIn[usPos + 1] == (PREAMBULE_PACKET & 0x00ff)))
578 break;
579 }
580
581 // Copy data in buffer whole
582 if ((pucPacketIn[usPos + 2] == TYPE_PACKET.CMD_READ_SETTINGS_FROM_FLASH) ||
583 (pucPacketIn[usPos + 2] == TYPE_PACKET.CMD_READ_ADDITIONAL_SETTINGS) ||
584 (pucPacketIn[usPos + 2] == TYPE_PACKET.CMD_READ_SETTINGS_FROM_RAM) || (pucPacketIn[usPos + 2] == TYPE_PACKET.CMD_RESET) ||
585 (pucPacketIn[usPos + 2] == TYPE_PACKET.CMD_START) || (pucPacketIn[usPos + 2] == TYPE_PACKET.CMD_STOP) ||
586 (pucPacketIn[usPos + 2] == TYPE_PACKET.PING_ID) ||
