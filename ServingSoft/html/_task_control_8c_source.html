262void GetTargetOneWheel (TApiControlTargetValue *target, const TProtocolSettingsWheel * restrict wheel, TEnumIdWheel which, float time)
263{
264 static float fPrevTargetAngularLeft, fPrevTargetAngularRight; // Для автокалибровки
265 static float fPrevTargetCalibrRightBack=0.f,fPrevTargetCalibrRightForward=0.f,fPrevTargetCalibrLeftBack=0.f,fPrevTargetCalibrLeftForward=0.f;
266 static TEnumExtremumCombination xEnumMinToMaxCombinationLeft = MIN_TO_MAX_TO_MIN_ASSIGNED;
267 static TEnumExtremumCombination xEnumMaxToMinCombinationLeft = MIN_TO_MAX_TO_MIN_ASSIGNED;
268 static TEnumExtremumCombination xEnumMinToMaxCombinationRight = MIN_TO_MAX_TO_MIN_ASSIGNED;
269 static TEnumExtremumCombination xEnumMaxToMinCombinationRight = MIN_TO_MAX_TO_MIN_ASSIGNED;
270
271 assert_param(target != NULL);
272 assert_param(wheel != NULL);
273 assert_param(time >= 0.f);
274
275 // if the control is off, there is no target
276 if (controlState == STATE_CONTROL_OFF)
277 {
278 target->left = target->right = TARGET_Q;
279 target->valueLeft = target->valueRight = 0;
280
281 xEnumMinToMaxCombinationLeft = xEnumMaxToMinCombinationLeft = MIN_TO_MAX_TO_MIN_ASSIGNED;
282 return;
283 }
284
285
286 // Using regulator
287 if ((wheel->usingRegulator == PID_REGULATOR) || (wheel->usingRegulator == LINEAR_REGULATOR)) {
288 // Const mode
289 if ((wheel->settingWithRegulator.regulatorObjective == OBJECTIVE_W) && (wheel->settingWithRegulator.modeSignal == MODE_CONST)) {
290 if (which == ID_LEFT_WHEEL) {
291 target->left = TARGET_W;
292 target->valueLeft = wheel->settingWithRegulator.valueConstW;
293 } else {
294 target->right = TARGET_W;
295 target->valueRight = (wheel->settingWithRegulator.valueConstW);
296 }
297 }
298 if ((wheel->settingWithRegulator.regulatorObjective == OBJECTIVE_Q) && (wheel->settingWithRegulator.modeSignal == MODE_CONST)) {
299 if (which == ID_LEFT_WHEEL) {
300 target->left = TARGET_Q;
301 target->valueLeft = xControlSettingRamMutex.settingControl->wheelLeft.settingWithRegulator.valueConstQ + interfaceResetCondition.angularLeft;
302 } else {
303 target->right = TARGET_Q;
304 target->valueRight = (wheel->settingWithRegulator.valueConstQ + interfaceResetCondition.angularLeft);
305 }
306 }
307 // Sinus mode
308 if ((wheel->settingWithRegulator.regulatorObjective == OBJECTIVE_W) && (wheel->settingWithRegulator.modeSignal == MODE_SIN)) {
309 if (which == ID_LEFT_WHEEL) {
310 target->left = TARGET_W;
311 target->valueLeft = wheel->settingWithRegulator.valueAmplitudeW * sin((2.0*PI * time) / wheel->settingWithRegulator.valuePeriodW);
312 } else {
313 target->right = TARGET_W;
314 target->valueRight = (wheel->settingWithRegulator.valueAmplitudeW * sin((2.0*PI * time) / wheel->settingWithRegulator.valuePeriodW));
315 }
316 }
317 if ((wheel->settingWithRegulator.regulatorObjective == OBJECTIVE_Q) && (wheel->settingWithRegulator.modeSignal == MODE_SIN))
318 {
319 if (which == ID_LEFT_WHEEL)
320 {
321 target->left = TARGET_Q;
322 target->valueLeft = wheel->settingWithRegulator.valueAmplitudeQ * sin((2.0*PI * time) / wheel->settingWithRegulator.valuePeriodQ);
323 // Value of target angle from min to max
324 if (target->valueLeft > fPrevTargetAngularLeft)
325 {
326 fPrevTargetAngularLeft = target->valueLeft; // To determine the point of extremum
327 // Switch to a new coefficient only when we have passed the minimum
328 xCalibrAmplitudeCoef.leftForwardCurrent = xCalibrAmplitudeCoef.leftForwardNew;
329 // Calculation a new calibration target value
330/***/ if (target->valueLeft < 0.f) {
331 fPrevTargetCalibrLeftForward = target->valueLeft + target->valueLeft*(1.f - xCalibrAmplitudeCoef.leftForwardCurrent);
332 } else {
333 fPrevTargetCalibrLeftForward = target->valueLeft * xCalibrAmplitudeCoef.leftForwardCurrent;
334 }
335 // if we are the minimum point, then determine the combination
336 if ((xEnumMinToMaxCombinationLeft == MIN_TO_MAX_TO_MIN_ASSIGNED) && (uCountCycleNotAutocalibr >= COUNT_CYCLES_NOT_AUTOCALIBRATION))
337 {
338 xEnumMaxToMinCombinationLeft = MIN_TO_MAX_TO_MIN_ASSIGNED; // Reset combination of maximum point
339 if (condWheelLeft.currentAngular > fPrevTargetCalibrLeftForward)
340 {
341 xEnumMinToMaxCombinationLeft = MIN_TO_MAX__B_R_F__R_B_F__R_F_B;
342 } else {
343 xEnumMinToMaxCombinationLeft = MIN_TO_MAX_ONLY_F;
344 }
345 }
346 // Depending on the condition, determine the target
347 switch (xEnumMinToMaxCombinationLeft)
348 {
349 case MIN_TO_MAX__B_R_F__R_B_F__R_F_B:
350/***/ if (fPrevTargetCalibrLeftBack > fPrevTargetCalibrLeftForward) {
351 target->valueLeft = fPrevTargetCalibrLeftBack;//condWheelLeft.currentAngular;
352 } else {
353 target->valueLeft = fPrevTargetCalibrLeftForward;
354 xEnumMinToMaxCombinationLeft = MIN_TO_MAX_ONLY_F;
355 }
356 break;
357 case MIN_TO_MAX_ONLY_F:
358 target->valueLeft = fPrevTargetCalibrLeftForward;
359 break;
360 case MIN_TO_MAX_TO_MIN_ASSIGNED:
361 target->valueLeft = target->valueLeft;
362 break;
363 default:
364 target->valueLeft = target->valueLeft;
365 xEnumMinToMaxCombinationLeft = MIN_TO_MAX_TO_MIN_ASSIGNED;
366 break;
367 }
368 // Value of target angle from max to min
369 } else {
370 fPrevTargetAngularLeft = target->valueLeft; // To determine the point of extremum
371 // Switch to a new coefficient only when we have passed max
372 xCalibrAmplitudeCoef.leftBackCurrent = xCalibrAmplitudeCoef.leftBackNew;
373 // Calculation a new calibration target value
374 if (target->valueLeft > 0.f) {
375 fPrevTargetCalibrLeftBack = target->valueLeft + target->valueLeft * (1 - xCalibrAmplitudeCoef.leftBackCurrent);
376 } else {
377 fPrevTargetCalibrLeftBack = target->valueLeft * xCalibrAmplitudeCoef.leftBackCurrent;
378 }
379 // if we are the maximum point, then determine the combination
380 if ((xEnumMaxToMinCombinationLeft == MIN_TO_MAX_TO_MIN_ASSIGNED) && (uCountCycleNotAutocalibr >= COUNT_CYCLES_NOT_AUTOCALIBRATION))
381 {
382 xEnumMinToMaxCombinationLeft = MIN_TO_MAX_TO_MIN_ASSIGNED; // Reset combination of minimum point
383 if (fPrevTargetCalibrLeftBack > condWheelLeft.currentAngular)
384 {
385 xEnumMaxToMinCombinationLeft = MAX_TO_MIN__B_R_F__B_F_R__F_B_R;
386 } else {
387 xEnumMaxToMinCombinationLeft = MAX_TO_MIN_ONLY_B;
388 }
389 }
390 // Depending on the condition, determine the target
391 switch (xEnumMaxToMinCombinationLeft)
392 {
393 case MAX_TO_MIN__B_R_F__B_F_R__F_B_R:
394 if (fPrevTargetCalibrLeftBack > fPrevTargetCalibrLeftForward) {
395 target->valueLeft = fPrevTargetCalibrLeftForward;
396 } else {
397 target->valueLeft = fPrevTargetCalibrLeftBack;
398 xEnumMaxToMinCombinationLeft = MAX_TO_MIN_ONLY_B;
399 }
400 break;
401 case MAX_TO_MIN_ONLY_B:
402 target->valueLeft = fPrevTargetCalibrLeftBack;
403 break;
404 case MIN_TO_MAX_TO_MIN_ASSIGNED:
405 target->valueLeft = target->valueLeft;
406 break;
407 default:
408 target->valueLeft = condWheelLeft.currentAngular;
409 xEnumMaxToMinCombinationLeft = MIN_TO_MAX_TO_MIN_ASSIGNED;
410 break;
411 }
412 }
413 } else {
414 // RIGHT WHEEL
415 target->right = TARGET_Q;
416 target->valueRight = wheel->settingWithRegulator.valueAmplitudeQ * sin((2.0*PI * time) / wheel->settingWithRegulator.valuePeriodQ);
417 // Value of target angle from min to max
418 if (target->valueRight > fPrevTargetAngularRight)
419 {
420 fPrevTargetAngularRight = target->valueRight; // To determine the point of extremum
421 // Switch to a new coefficient only when we have passed the minimum
422 xCalibrAmplitudeCoef.rightForwardCurrent = xCalibrAmplitudeCoef.rightForwardNew;
423 // Calculation a new calibration target value
424 if (target->valueRight < 0.f) {
425 fPrevTargetCalibrRightForward = target->valueRight + target->valueRight*(1.f - xCalibrAmplitudeCoef.rightForwardCurrent);
426 } else {
427 fPrevTargetCalibrRightForward = target->valueRight * xCalibrAmplitudeCoef.rightForwardCurrent;
428 }
429 // if we are the minimum point, then determine the combination
430 if ((xEnumMinToMaxCombinationRight == MIN_TO_MAX_TO_MIN_ASSIGNED) && (uCountCycleNotAutocalibr >= COUNT_CYCLES_NOT_AUTOCALIBRATION))
431 {
432 xEnumMaxToMinCombinationRight = MIN_TO_MAX_TO_MIN_ASSIGNED; // Reset combination of maximum point
433 if (condWheelRight.currentAngular > fPrevTargetCalibrRightForward)
434 {
435 xEnumMinToMaxCombinationRight = MIN_TO_MAX__B_R_F__R_B_F__R_F_B;
436 } else {
437 xEnumMinToMaxCombinationRight = MIN_TO_MAX_ONLY_F;
438 }
439 }
440 // Depending on the condition, determine the target
441 switch (xEnumMinToMaxCombinationRight)
442 {
443 case MIN_TO_MAX__B_R_F__R_B_F__R_F_B:
444 if (fPrevTargetCalibrRightBack > fPrevTargetCalibrRightForward) {
445 target->valueRight = fPrevTargetCalibrRightBack;
446 } else {
447 target->valueRight = fPrevTargetCalibrRightForward;
448 xEnumMinToMaxCombinationRight = MIN_TO_MAX_ONLY_F;
449 }
450 break;
451 case MIN_TO_MAX_ONLY_F:
452 target->valueRight = fPrevTargetCalibrRightForward;
453 break;
454 case MIN_TO_MAX_TO_MIN_ASSIGNED:
455 target->valueRight = target->valueRight;
456 break;
457 default:
458 target->valueRight = condWheelRight.currentAngular;
459 xEnumMinToMaxCombinationRight = MIN_TO_MAX_TO_MIN_ASSIGNED;
460 break;
461 }
462 // Value of target angle from max to min
463 } else {
464 fPrevTargetAngularRight = target->valueRight; // To determine the point of extremum
465 // Switch to a new coefficient only when we have passed max
466 xCalibrAmplitudeCoef.rightBackCurrent = xCalibrAmplitudeCoef.rightBackNew;
467 // Calculation a new calibration target value
468 if (target->valueRight > 0.f) {
469 fPrevTargetCalibrRightBack = target->valueRight + target->valueRight * (1 - xCalibrAmplitudeCoef.rightBackCurrent);
470 } else {
471 fPrevTargetCalibrRightBack = target->valueRight * xCalibrAmplitudeCoef.rightBackCurrent;
472 }
473 // if we are the maximum point, then determine the combination
474 if ((xEnumMaxToMinCombinationRight == MIN_TO_MAX_TO_MIN_ASSIGNED) && (uCountCycleNotAutocalibr >= COUNT_CYCLES_NOT_AUTOCALIBRATION))
475 {
476 xEnumMinToMaxCombinationRight = MIN_TO_MAX_TO_MIN_ASSIGNED; // Reset combination of minimum point
477 if (fPrevTargetCalibrRightBack > condWheelRight.currentAngular)
478 {
479 xEnumMaxToMinCombinationRight = MAX_TO_MIN__B_R_F__B_F_R__F_B_R;
480 } else {
481 xEnumMaxToMinCombinationRight = MAX_TO_MIN_ONLY_B;
482 }
483 }
484 // Depending on the condition, determine the target
485 switch (xEnumMaxToMinCombinationRight)
486 {
487 case MAX_TO_MIN__B_R_F__B_F_R__F_B_R:
488 if (fPrevTargetCalibrRightBack > fPrevTargetCalibrRightForward) {
489 target->valueRight = fPrevTargetCalibrRightForward;
490 } else {
491 target->valueRight = fPrevTargetCalibrRightBack;
492 xEnumMaxToMinCombinationRight = MAX_TO_MIN_ONLY_B;

507 }
508 }
509 }
510
511 // Without regulator
512 if (wheel->usingRegulator == WITHOUT_REGULATOR) {
513 if (which == ID_LEFT_WHEEL) {

605{
606 uint16_t it = 0;
607 static TEnumStateControl prevStateControl;
608
609 /* USER CODE BEGIN FuncTaskControl */
610
611 FunctionTayControl(true);
612 /* Infinite loop */
613 for(;;)
614 {
615 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
616
617 // Because this task has the highest priority and can be called then the coefficients being update
618 while (xSemaphoreTake(xControlSettingRamMutex.xMutexThis, 0) == pdFALSE)
619 {
620 if (it++ > 0xFFF0) {
621 osDelay(10);
622 }
623 }
624
625 // Using regulator always
626 if (FunctionTayControl(false) == API_RESULT_FAIL) {
627 xSemaphoreGive(xControlSettingRamMutex.xMutexThis);
628 terminalPrintf("Error in TAY function");

630 } else {
631 // if not using regulator - then take duty cycle from RAM setting
632 if (xControlSettingRamMutex.settingControl->wheelLeft.usingRegulator == WITHOUT_REGULATOR) {
633 duty_w_queue.dutyCycleLeft = xControlSettingRamMutex.settingControl->wheelLeft.settingsWithoutRegulator.valueDutyCycle;
634 duty_w_queue.dutyCycleRight = xControlSettingRamMutex.settingControl->wheelRight.settingsWithoutRegulator.valueDutyCycle;
635 }
636 xSemaphoreGive(xControlSettingRamMutex.xMutexThis);
637 // if the command to start control has been accepted
638 if ((prevStateControl == STATE_CONTROL_OFF) && (controlState == STATE_CONTROL_ON)) {
639 // Find current hall sensor condition pointers
640 // TODO UNCOMMENT!!! ONLY FOR DEBUG WITHOUT STROLLER !!!!!!
641#if DEBUG_MODE_WITHOUT_HALL_SENSOR == 0

643 while (funcPhyLevel.Goto(&condWheelRight, funcPhyLevel.GetConditionHall(&condWheelRight)) == NO_VALID_HALL_CONDITION) {};
644#endif
645 funcPhyLevel.SetDutyCycleLeft(&condWheelLeft, duty_w_queue.dutyCycleLeft); /***/ //Below duty_w_queue adding in xQueueTaskControlToTaskPhy
646 funcPhyLevel.SetDutyCycleRight(&condWheelRight, duty_w_queue.dutyCycleRight); /***/ // May be it is unnecessary
647 funcPhyLevel.SetKeysLeft(&condWheelLeft);
648 funcPhyLevel.SetKeysRight(&condWheelRight);
649 FunctionTayControl(true);
650 }
651 prevStateControl = controlState;
652 // if control off
653 if (controlState == STATE_CONTROL_OFF) {
654 duty_w_queue.dutyCycleLeft = 0;
655 duty_w_queue.dutyCycleRight = 0;
656 }
657
658 /***/
659 // Подстраховка на случай подвиса
660 // uint16_t leftHall1, leftHall2, leftHall3;
661 // uint16_t rightHall1, rightHall2, rightHall3;
662 // leftHall1 = leftHall2 = leftHall3 = rightHall1 = rightHall2 = rightHall3 = 0;
663 uint8_t conditionLeft, conditionRight;
664 conditionLeft = GetConditionHall(&condWheelLeft);
665 conditionRight = GetConditionHall(&condWheelRight);
666 // leftHall1 = (condWheelLeft.xHallSensor.hall_1.port->IDR & condWheelLeft.xHallSensor.hall_1.pin);
667 // leftHall2 = (condWheelLeft.xHallSensor.hall_2.port->IDR & condWheelLeft.xHallSensor.hall_2.pin);
668 // leftHall3 = (condWheelLeft.xHallSensor.hall_3.port->IDR & condWheelLeft.xHallSensor.hall_3.pin);
