165void FuncTaskControl (void *argument) // @suppress("Name convention for function")
166{
167 uint16_t it = 0;
168 static TEnumStateControl xPrevStateControl_;
169
170 /* USER CODE BEGIN FuncTaskControl */
171 FunctionTayControl(true);
172 /* Infinite loop */
173 for(;;)
174 {
175 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
176
177 // Because this task has the highest priority and can be called then the coefficients being update
178 while (xSemaphoreTake(xControlSettingRamMutex.xMutexThis, 0) == pdFALSE)
179 {
180 if (it++ > 0xFFF0)
181 osDelay(10);
182 }
183
184 // Using regulator always
185 if (FunctionTayControl(false) == API_RESULT_FAIL)
186 {
187 xSemaphoreGive(xControlSettingRamMutex.xMutexThis);
188 xFuncInterfaceLevel.xTerminalSend.ePrintf("Error in TAY function");
189 goto nextIterationFuncTaskControl;
190 }
191
192 // if not using regulator - then take duty cycle from RAM setting
193 if (xControlSettingRamMutex.pxSettingControl->wheelLeft.usingRegulator == WITHOUT_REGULATOR)
194 {
195 xDutyW_Queue_.fDutyCycleLeft = xControlSettingRamMutex.pxSettingControl->wheelLeft.settingsWithoutRegulator.valueDutyCycle;
196 xDutyW_Queue_.fDutyCycleRight = xControlSettingRamMutex.pxSettingControl->wheelRight.settingsWithoutRegulator.valueDutyCycle;
197 }
198 xSemaphoreGive(xControlSettingRamMutex.xMutexThis);
199
200 // if the command to start control has been accepted
201 if ((xPrevStateControl_ == STATE_CONTROL_OFF) && (eControlState == STATE_CONTROL_ON))
202 {
203 prvStartMoving();
204 }
205 xPrevStateControl_ = eControlState;
206
207 // if control off
208 if (eControlState == STATE_CONTROL_OFF)
209 {
210 xDutyW_Queue_.fDutyCycleLeft = 0;
211 xDutyW_Queue_.fDutyCycleRight = 0;
212 }
213
214 // Подстраховка на случай подвиса
215 prvWheelLockSafety();
216
217 // Send message to other tasks
218 xQueueOverwrite(xQueueTaskControlToTaskPhy, &xDutyW_Queue_);
219 xTaskNotifyGive(TaskPhyLevelHandle);
220 xQueueOverwrite(xQueueTaskControlToTaskInterface, &xDutyW_Queue_);
221
222nextIterationFuncTaskControl:
223 (void)0;
224 }
225 /* USER CODE END FuncTaskControl */
226}

234static void prvTranslationRotationControl (float fPhaseQ)
235{
236 float fSetpointAmplQ = xControlSettingRamMutex.pxSettingControl->wheelLeft.settingWithRegulator.valueAmplitudeQ; // Реальная амплитуда из Пульта
237
238 assert_param((fPhaseQ >= 0.f) && (fPhaseQ <= 2.f*PI));
239
240 // Фаза ноль и движение к максимуму (0 рад)
241 if ((eZeroToMaxWait_Translation_ == EVENT_EXPECTED) && (fPhaseQ < PI/4.f))
242 {
243 prvCalculateSetpointsInPhase_0_AndMoveToMax(fSetpointAmplQ);
244 }
245 // В максимуме (0.5*Pi)
246 if ( (eMaxWait_Translation_ == EVENT_EXPECTED) && ((fPhaseQ >= 0.5*PI) && (fPhaseQ < PI)) )
247 {
248 prvCalculateSetpointsInPhase_HalfPi_Maximum(fSetpointAmplQ);
249 }
250 // Фаза 1.0*Pi
251 if ( (eZeroToMinWait_Translation_ == EVENT_EXPECTED) && ((fPhaseQ >= PI) && (fPhaseQ < (1.5)*PI)) )
252 {
253 prvCalculateSetpointsInPhase_Pi_AndMoveToMin(fSetpointAmplQ);
254 }
255 // В минимуме (1.5*Pi)
256 if ((eMinWait_Translation_ == EVENT_EXPECTED) && (fPhaseQ >= 1.5*PI))
257 {
258 prvCalculateSetpointsInPhase_OneAndHalfPi_Minimum(fSetpointAmplQ);
259 }

305static TEnumApiControlResult prvExecuteMotionCommand (uint8_t cmd, float fTimeControl, float fPeriodQ,
306 float fPhaseCalibr, float fSetpoint)
307{
308 TEnumApiControlResult ret = API_RESULT_OK;
309
310 if (xSemaphoreTakeRecursive(xControlSettingRamMutex.xMutexThis, 0) == pdFALSE)
311 {
312 ret = API_RESULT_FAIL;
313 goto return_prvExecuteMotionCommand;
314 }
315
316 switch (cmd)
317 {
318 float fCoef;
319 case CMD_REAL_INC_COEF_WHEEL_ROTATION_DEFINE :
320 fCoefWheel_Rotation_ = fCoefWheel_Rotation_ + STEP_CALIBR_Q_ROTATION_;
321 xFuncInterfaceLevel.xTerminalSend.eCoefWheelRotation(fCoefWheel_Rotation_);
322 break;
323 case CMD_REAL_DEC_COEF_WHEEL_ROTATION_DEFINE :
324 fCoefWheel_Rotation_ = fCoefWheel_Rotation_ - STEP_CALIBR_Q_ROTATION_;
325 xFuncInterfaceLevel.xTerminalSend.eCoefWheelRotation(fCoefWheel_Rotation_);
326 break;
327 case CMD_REAL_INC_COEF_WHEEL_TRANSLATION_DEFINE :
328 fCoefWheel_Translation_ = fCoefWheel_Translation_ + STEP_CALIBR_Q_TRANSLATION_;
329 xFuncInterfaceLevel.xTerminalSend.eCoefWheelTranslation(fCoefWheel_Translation_);
330 break;
331 case CMD_REAL_DEC_COEF_WHEEL_TRANSLATION_DEFINE :
332 fCoefWheel_Translation_ = fCoefWheel_Translation_ - STEP_CALIBR_Q_TRANSLATION_;
333 xFuncInterfaceLevel.xTerminalSend.eCoefWheelTranslation(fCoefWheel_Translation_);
334 break;
335 case CMD_REAL_INC_AMPLITUDE_DEFINE :
336 fCoef = 1.f + STEP_CALIBR_AMPLITUDE_;
337 xControlSettingRamMutex.pxSettingControl->wheelLeft.settingWithRegulator.valueAmplitudeQ =
338 xControlSettingRamMutex.pxSettingControl->wheelRight.settingWithRegulator.valueAmplitudeQ =
339 xControlSettingRamMutex.pxSettingControl->wheelLeft.settingWithRegulator.valueAmplitudeQ * fCoef;
340 xFuncInterfaceLevel.xTerminalSend.eAmpliude(
341 xControlSettingRamMutex.pxSettingControl->wheelLeft.settingWithRegulator.valueAmplitudeQ);
342 break;
343 case CMD_REAL_DEC_AMPLITUDE_DEFINE :
344 fCoef = 1.f - STEP_CALIBR_AMPLITUDE_;
345 xControlSettingRamMutex.pxSettingControl->wheelLeft.settingWithRegulator.valueAmplitudeQ =
346 xControlSettingRamMutex.pxSettingControl->wheelRight.settingWithRegulator.valueAmplitudeQ =
347 xControlSettingRamMutex.pxSettingControl->wheelLeft.settingWithRegulator.valueAmplitudeQ * fCoef;
348 xFuncInterfaceLevel.xTerminalSend.eAmpliude(
349 xControlSettingRamMutex.pxSettingControl->wheelLeft.settingWithRegulator.valueAmplitudeQ);
350 break;
351 case CMD_REAL_INC_PERIOD_DEFINE :
352 fCoef = 1.f + STEP_CALIBR_PERIOD_;
353 fPeriodQ = fPeriodQ * fCoef;
354 dPhaseQ_ = 2.f*PI*(fmodf(fTimeControl, fPeriodQ)/fPeriodQ) - fPhaseCalibr;
355 xControlSettingRamMutex.pxSettingControl->wheelLeft.settingWithRegulator.valuePeriodQ =
356 xControlSettingRamMutex.pxSettingControl->wheelRight.settingWithRegulator.valuePeriodQ =
357 xControlSettingRamMutex.pxSettingControl->wheelLeft.settingWithRegulator.valuePeriodQ * fCoef;
358 xFuncInterfaceLevel.xTerminalSend.ePeriod(
359 xControlSettingRamMutex.pxSettingControl->wheelLeft.settingWithRegulator.valuePeriodQ);
360 break;
361 case CMD_REAL_DEC_PERIOD_DEFINE :
362 fCoef = 1.f - STEP_CALIBR_PERIOD_;
363 fPeriodQ = fPeriodQ * fCoef;
364 dPhaseQ_ = 2.f*PI*(fmodf(fTimeControl, fPeriodQ)/fPeriodQ) - fPhaseCalibr;
365 xControlSettingRamMutex.pxSettingControl->wheelLeft.settingWithRegulator.valuePeriodQ =
366 xControlSettingRamMutex.pxSettingControl->wheelRight.settingWithRegulator.valuePeriodQ =
367 xControlSettingRamMutex.pxSettingControl->wheelLeft.settingWithRegulator.valuePeriodQ * fCoef;
368 xFuncInterfaceLevel.xTerminalSend.ePeriod(
369 xControlSettingRamMutex.pxSettingControl->wheelLeft.settingWithRegulator.valuePeriodQ);
370 break;
371 case CMD_REAL_FORWARD_DEFINE :
372 fStepTrackForward_ = fStepTrackForward_ + fSetpoint*STEP_CALIBR_MOVE_;
373 xFuncInterfaceLevel.xTerminalSend.ePrintf("The stroller will move forward");
374 break;
375 case CMD_REAL_BACK_DEFINE :
376 fStepTrackBack_ = fStepTrackBack_ + fSetpoint*STEP_CALIBR_MOVE_;
377 xFuncInterfaceLevel.xTerminalSend.ePrintf("The stroller will move back");
378 break;
379 default:
380 ret = API_RESULT_FAIL;
381 break;
382 }
383
384return_prvExecuteMotionCommand:
385 xSemaphoreGiveRecursive(xControlSettingRamMutex.xMutexThis);
386 return ret;
387}

474static void prvCalculateSetpointsInPhase_HalfPi_Maximum (float fSetpointAmplQ)
475{
476 float dQ = xConditionWheelLeft.fCurrentAngular - xConditionWheelRight.fCurrentAngular;
477
478 eMaxWait_Translation_ = EVENT_BEEN;
479 eZeroToMinWait_Translation_ = EVENT_EXPECTED;
480 // Выравнивание углов колёс с изменением уставки: чтобы не было скачка
481 xConditionWheelRight.fCurrentAngular = xConditionWheelLeft.fCurrentAngular;
482 fAmplForwardRightQ_RotationTranslation_ = fAmplForwardRightQ_RotationTranslation_ + dQ;
483 // Если двигали коляску вперёд по команде
484 xConditionWheelLeft.fCurrentAngular = xConditionWheelLeft.fCurrentAngular - fPrevStepTrackForward_;
485 xConditionWheelRight.fCurrentAngular = xConditionWheelRight.fCurrentAngular - fPrevStepTrackForward_;
486 fAmplForwardLeftQ_RotationTranslation_ = fAmplForwardLeftQ_RotationTranslation_ - fPrevStepTrackForward_;
487 fAmplForwardRightQ_RotationTranslation_ = fAmplForwardRightQ_RotationTranslation_ - fPrevStepTrackForward_;
488 // Зафиксировали уставку в данный момент
489 fFixSetpointLeft_RotationTranslation_ = fAmplForwardLeftQ_RotationTranslation_; // = fSetpointAmplQ
490 fFixSetpointRight_RotationTranslation_ = fAmplForwardRightQ_RotationTranslation_; // = fSetpointAmplQ*Kr+dQ
491 // Уставка до Pi радиан
492 float fDistanceSPBackL = (fCoefWheel_Rotation_ >= 0.f) ?
493 fAmplForwardLeftQ_RotationTranslation_ + fSetpointAmplQ * (1.f - fCoefWheel_Rotation_) :
494 fAmplForwardLeftQ_RotationTranslation_+fSetpointAmplQ;
495 float fDistanceSPBackR = (fCoefWheel_Rotation_ >= 0.f) ?
496 fAmplForwardRightQ_RotationTranslation_ + fSetpointAmplQ :
497 fAmplForwardRightQ_RotationTranslation_ + fSetpointAmplQ * (1.f + fCoefWheel_Rotation_);
498 fDistanceSPBackL = fDistanceSPBackL + fStepTrackBack_;// Если была команда сдвинуть коляску назад
499 fDistanceSPBackR = fDistanceSPBackR + fStepTrackBack_;
500 fAmplBackLeftQ_RotationTranslation_ = fAmplForwardLeftQ_RotationTranslation_ -
501 (fDistanceSPBackL / 2.f) * (1.f + fCoefWheel_Translation_);
502 fAmplBackRightQ_RotationTranslation_ = fAmplForwardRightQ_RotationTranslation_ -
503 (fDistanceSPBackR / 2.f) * (1.f - fCoefWheel_Translation_);
504}

542static void prvCalculateSetpointsInPhase_OneAndHalfPi_Minimum (float fSetpointAmplQ)
543{
544 eMinWait_Translation_ = EVENT_BEEN;
545 eZeroToMaxWait_Translation_ = EVENT_EXPECTED;
546 // Выравнивание углов колёс с изменением уставки: чтобы не было скачка
547 float dQ = xConditionWheelRight.fCurrentAngular - xConditionWheelLeft.fCurrentAngular;
548 xConditionWheelLeft.fCurrentAngular = xConditionWheelRight.fCurrentAngular;
549 fAmplBackLeftQ_RotationTranslation_ = fAmplBackLeftQ_RotationTranslation_ + dQ;
550 // Если двигали коляску назад по команде
551 xConditionWheelLeft.fCurrentAngular = xConditionWheelLeft.fCurrentAngular + fStepTrackBack_;
552 xConditionWheelRight.fCurrentAngular = xConditionWheelRight.fCurrentAngular + fStepTrackBack_;
553 fAmplBackLeftQ_RotationTranslation_ = fAmplBackLeftQ_RotationTranslation_ + fStepTrackBack_;
554 fAmplBackRightQ_RotationTranslation_ = fAmplBackRightQ_RotationTranslation_ + fStepTrackBack_;
555 // Зафиксировали уставку в данный момент
556 fFixSetpointLeft_RotationTranslation_ = fAmplBackLeftQ_RotationTranslation_; // = -fSetpointAmplQ*Kr+dQ
557 fFixSetpointRight_RotationTranslation_ = fAmplBackRightQ_RotationTranslation_; // = -fSetpointAmplQ
558 // Уставка до 2*Pi радиан
559 float fDistanceSPForwardL = (fCoefWheel_Rotation_ >= 0.f) ?
560 (-1.f) * fAmplBackLeftQ_RotationTranslation_ + fSetpointAmplQ :
561 (-1.f) * fAmplBackLeftQ_RotationTranslation_ + fSetpointAmplQ * (1.f + fCoefWheel_Rotation_);
562 float fDistanceSPForwardR = (fCoefWheel_Rotation_ >= 0.f) ?
563 (-1.f) * fAmplBackRightQ_RotationTranslation_ + fSetpointAmplQ * (1.f - fCoefWheel_Rotation_) :
564 (-1.f) * fAmplBackRightQ_RotationTranslation_ + fSetpointAmplQ;
565 fDistanceSPForwardL = fDistanceSPForwardL + fStepTrackForward_;// Если была команда сдвинуть коляску назад
566 fDistanceSPForwardR = fDistanceSPForwardR + fStepTrackForward_;
567 fAmplForwardLeftQ_RotationTranslation_ = fAmplBackLeftQ_RotationTranslation_ +
568 (fDistanceSPForwardL / 2.f) * (1.f + fCoefWheel_Translation_);
569 fAmplForwardRightQ_RotationTranslation_ = fAmplBackRightQ_RotationTranslation_ +
570 (fDistanceSPForwardR / 2.f) * (1.f - fCoefWheel_Translation_);
571}

862

883 {
884 xEventCmdExecution_ = EVENT_BEEN;
885 fPrevStepTrackForward_ = fStepTrackForward_; // В prvTranslationRotation в точке PI/2 пригодится
886 fStepTrackForward_ = fStepTrackBack_ = 0.f;
