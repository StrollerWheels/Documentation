226{
227 uint8_t pucQueueTemp[ ((sizeof(TDutyCycleAngularVelocity)>sizeof(TEventHall))?sizeof(TDutyCycleAngularVelocity):sizeof(TEventHall)) ];
228
229 // Task function (infinite loop)
230 for(;;)
231 {
232 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
233
234 // if there is unread data from the taskControl
235 if (xQueueReceive(xQueueTaskControlToTaskPhy, pucQueueTemp, 0) == pdTRUE)

240 xConditionWheelLeft.fCurrAngularVelocity = ((TDutyCycleAngularVelocity*)(pucQueueTemp))->fAngularVelocityLeft;

243 xConditionWheelRight.fCurrAngularVelocity = ((TDutyCycleAngularVelocity*)(pucQueueTemp))->fAngularVelocityRight;
244 xSemaphoreGive(xConditionWheelRight.xMutexThis);
245 }
246 // if there is unread data from the Interrupt of Hall sensor
247 while (xQueueReceive(xQueueWheelToTaskPhy, pucQueueTemp, 0) == pdTRUE)
248 {
249 if (((TEventHall*) (pucQueueTemp))->eNoWheel == ID_LEFT_WHEEL)
250 {

256 prvSafetyEventProcessing(&xConditionWheelLeft);
257#endif
258 }
259 if (((TEventHall*) (pucQueueTemp))->eNoWheel == ID_RIGHT_WHEEL)
260 {
261 if (eControlState == STATE_CONTROL_ON)
262 prvHallSensorsEventSinusoidalProcessing (&xConditionWheelRight, pucQueueTemp);
263 prvHallSensorsEventProcessing(&xConditionWheelRight, pucQueueTemp, ID_RIGHT_WHEEL);
264#if SAFETY_USE == 1
265 if (eControlState == STATE_CONTROL_ON)
266 prvSafetyEventProcessing(&xConditionWheelRight);
267#endif
268 }
269 }
270 }

287static void prvHallSensorsEventProcessing (TConditionWheelPhyLevel *pxConditionWheel, uint8_t *queue, TEnumIdWheel eWheel)
288{
289 int8_t countStep; // Поворот колеса на количетсво комбинаций Холла
290 TAngularEvent angularEvent; // для передачи в задачу управления
291 TAngularEvent tempEvent; // For receive from queue
292
293 assert_param(pxConditionWheel != NULL);
294 assert_param(queue != NULL);
295
296 countStep = xFuncPhyLevel.scGoto(pxConditionWheel, ((TEventHall*) (queue))->ucCondition);
297 switch (countStep)
298 {
299 case NO_VALID_HALL_CONDITION :
300 pxConditionWheel->usCountSlipHall++;
301 break;
302 case 1 :
303 pxConditionWheel->fCurrentAngular += countStep * RAD_IN_STEP_HALL;
304 break;
305 case -1 :
306 pxConditionWheel->fCurrentAngular += countStep * RAD_IN_STEP_HALL;
307 break;
308 default :
309 pxConditionWheel->usCountSlipHall++;
310 pxConditionWheel->fCurrentAngular += countStep * RAD_IN_STEP_HALL;
311 break;
312 }
313
314 if(countStep != NO_VALID_HALL_CONDITION)
315 {
316 if (eControlState == STATE_CONTROL_ON)
317 {
318 switch (pxConditionWheel->eIdWheel)
319 {
320 case ID_LEFT_WHEEL :
321 xFuncPhyLevel.vSetKeysLeft(pxConditionWheel);
322 ulTimeWithoutEventHall_Left_Ms_ = 0;
323 break;
324 case ID_RIGHT_WHEEL :
325 xFuncPhyLevel.vSetKeysRight(pxConditionWheel);
326 ulTimeWithoutEventHall_Right_Ms_ = 0;
327 break;
328 default :
329 assert_param(0);
330 break;
331 }
332 }
333
334 // Send event in queue to taskControl
335 angularEvent.fAngular = pxConditionWheel->fCurrentAngular;
336 angularEvent.eNoWheel = eWheel;
337 angularEvent.ullTimeEvent = ((TEventHall*) (queue))->ullTimeEvent + ullResetTime;
338 if (xQueueSendToBack(xQueueTaskPhyToTaskControl, &angularEvent, 0) == pdFALSE)
339 {
340 xQueueReceive(xQueueTaskPhyToTaskControl, &tempEvent, 0);
341 xQueueSendToBack(xQueueTaskPhyToTaskControl, &angularEvent, 0);
342 }
343 }
344}

357static void prvHallSensorsEventSinusoidalProcessing (TConditionWheelPhyLevel *pxConditionWheel, uint8_t *queue)
358{
359 assert_param(pxConditionWheel != NULL);
360 assert_param(queue != NULL);
361
362 // Number shift
363 pxConditionWheel->xSinusControlData.ucNoIntervalHallEventLast++;
364 if (pxConditionWheel->xSinusControlData.ucNoIntervalHallEventLast >= NUMBER_OF_COUNTABLE_INTERVAL_HALL_EVENT)
365 pxConditionWheel->xSinusControlData.ucNoIntervalHallEventLast = 0;
366
367 // Check in case there was reset command
368 for (size_t i = 0; i < NUMBER_OF_COUNTABLE_INTERVAL_HALL_EVENT; i++)
369 {
370 if ((((TEventHall*)queue))->ullTimeEvent < pxConditionWheel->xSinusControlData.pullHallEventLastTime_10mcs[i])
371 {
372 for (size_t j = 0; 0 < NUMBER_OF_COUNTABLE_INTERVAL_HALL_EVENT; j++)
373 {
374 pxConditionWheel->xSinusControlData.pullHallEventLastTime_10mcs[j] = ((TEventHall*)(queue))->ullTimeEvent;
375 }
376 pxConditionWheel->xSinusControlData.ucNoIntervalHallEventLast = 0;
377 break;
378 }
379 }
380
381 // Remember the hall event time
382 pxConditionWheel->xSinusControlData.pullHallEventLastTime_10mcs[pxConditionWheel->xSinusControlData.ucNoIntervalHallEventLast] =
383 ((TEventHall*)(queue))->ullTimeEvent;
384
385 // Calculation of event Hall period by moving averages
386 uint64_t ullMovingAvrSumTime = 0;
387 uint8_t currNo = pxConditionWheel->xSinusControlData.ucNoIntervalHallEventLast, prevNo;
388 for (size_t i = 0; i < NUMBER_OF_COUNTABLE_INTERVAL_HALL_EVENT - 1; i++)
389 {
390 if (currNo > 0)
391 {
392 prevNo = currNo - 1;
393 } else
394 {
395 prevNo = NUMBER_OF_COUNTABLE_INTERVAL_HALL_EVENT-1;
396 }
397 ullMovingAvrSumTime = ullMovingAvrSumTime + ( (pxConditionWheel->xSinusControlData.pullHallEventLastTime_10mcs[currNo] -
398 pxConditionWheel->xSinusControlData.pullHallEventLastTime_10mcs[prevNo]) * (NUMBER_OF_COUNTABLE_INTERVAL_HALL_EVENT - 1 - i) );
399 if (currNo == 0)
400 {
401 currNo = NUMBER_OF_COUNTABLE_INTERVAL_HALL_EVENT - 1;
402 } else {
403 currNo--;
404 }
405 }
406
407 // Moving average calculation
408 static const size_t devider_ = (NUMBER_OF_COUNTABLE_INTERVAL_HALL_EVENT * (NUMBER_OF_COUNTABLE_INTERVAL_HALL_EVENT - 1)) / 2;
409 pxConditionWheel->xSinusControlData.ulCurrentHallPeriod_10mcs = ullMovingAvrSumTime / devider_;
410
411 // Switching control mode if necessary
412 if ((pxConditionWheel->xAddSetting.typeControl == TRAPEZOIDAL_CONTROL) &&
413 (pxConditionWheel->eTypeModeControl_Current == SINUSOIDAL_CONTROL))
414 prvPhySwitchModeControl(pxConditionWheel, TRAPEZOIDAL_CONTROL);
415
416 if ((pxConditionWheel->xAddSetting.typeControl == SINUSOIDAL_CONTROL) &&
417 (pxConditionWheel->eTypeModeControl_Current == TRAPEZOIDAL_CONTROL) &&
418 (pxConditionWheel->xSinusControlData.ulCurrentHallPeriod_10mcs < HALL_EVENT_PERIOD_FOR_MODE_SINUS_ON_10MCS))
419 prvPhySwitchModeControl(pxConditionWheel, SINUSOIDAL_CONTROL);
420
421 if ((pxConditionWheel->xAddSetting.typeControl == SINUSOIDAL_CONTROL) &&
422 (pxConditionWheel->eTypeModeControl_Current == SINUSOIDAL_CONTROL) &&
423 (pxConditionWheel->xSinusControlData.ulCurrentHallPeriod_10mcs > HALL_EVENT_PERIOD_FOR_MODE_SINUS_OFF_10MCS))
424 prvPhySwitchModeControl(pxConditionWheel, TRAPEZOIDAL_CONTROL);
425}

434static bool prvSafetyEventProcessing (TConditionWheelPhyLevel *const pxConditionWheel)
435{
436 bool status = true;
437 float maxAmplitudeRad, maxAngularVelocityRadSec;
438
439 assert_param(pxConditionWheel != NULL);
440
441 maxAmplitudeRad =((2.f * pxConditionWheel->xAddSetting.maxAmplitude) / (100.f * WHEEL_D));
442 maxAngularVelocityRadSec = (pxConditionWheel->xAddSetting.maxLinearVelosity / WHEEL_R);
443
444 // Check amplitude of angular
445 if (fabsf(pxConditionWheel->fCurrentAngular) > maxAmplitudeRad)
446 {
447 xFuncPhyLevel.vCommandStateControl(CMD_BRAKE_CONTROL);
448 status = false;
449 xFuncInterfaceLevel.xTerminalSend.ePrintf("Exceeding the safe amplitude \n Braking was");
450 }
451
452 // Check difference between wheel angles
453 if (fabsf(xConditionWheelLeft.fCurrentAngular - xConditionWheelRight.fCurrentAngular) > pxConditionWheel->xAddSetting.maxDiffAngles)
454 {
455 xFuncPhyLevel.vCommandStateControl(CMD_BRAKE_CONTROL);
456 status = false;
457 xFuncInterfaceLevel.xTerminalSend.ePrintf("Exceeding safe difference wheel angles \n Braking was");
458 }
459
460 // Check velocity
461 if (fabsf(pxConditionWheel->fCurrAngularVelocity) > maxAngularVelocityRadSec)
462 {
463 xFuncPhyLevel.vCommandStateControl(CMD_BRAKE_CONTROL);
464 status = false;
465 xFuncInterfaceLevel.xTerminalSend.ePrintf("Exceeding safety angular velocity\n Braking was");
466 }
467
468 return status;
469}

593static float prvCalculationRealDutyCycle (TConditionWheelPhyLevel *pxConditionWheel, float fDutyCycle)
594{
595 float fDiff = fDutyCycle - pxConditionWheel->xSoftStartStop.fRealSoftDutyCycle;
596 float fRealDutyCycle;
597
598 assert_param(pxConditionWheel != NULL);
599
600 if ((fDiff > EPSILON) && (fDiff > pxConditionWheel->xSoftStartStop.fStepSoftDutyCycle))
601 fDiff = pxConditionWheel->xSoftStartStop.fStepSoftDutyCycle;
602
603 if ((fDiff < EPSILON) && (fabsf(fDiff) > pxConditionWheel->xSoftStartStop.fStepSoftDutyCycle))
604 fDiff = -(pxConditionWheel->xSoftStartStop.fStepSoftDutyCycle);
605
606 if (ullSystemTime_10us < pxConditionWheel->xSoftStartStop.ullPrevTimeChangePWM_10mcs)
607 pxConditionWheel->xSoftStartStop.ullPrevTimeChangePWM_10mcs = ullSystemTime_10us;
608
609 if ((ullSystemTime_10us - pxConditionWheel->xSoftStartStop.ullPrevTimeChangePWM_10mcs) >
610 pxConditionWheel->xSoftStartStop.ulPeriodSoftChange_10mcs)
611 {
612 fRealDutyCycle = pxConditionWheel->xSoftStartStop.fRealSoftDutyCycle + fDiff;
613 pxConditionWheel->xSoftStartStop.ullPrevTimeChangePWM_10mcs = ullSystemTime_10us;
614 } else {
615 fRealDutyCycle = pxConditionWheel->xSoftStartStop.fRealSoftDutyCycle;
616 }
617
618 return fRealDutyCycle;
619}

674static int8_t prvGotoValidHallCondition (TConditionWheelPhyLevel *pxConditionWheel, uint8_t ucCondition)
675{
676 int8_t findPtr, countStep;
677 int8_t ret = 0;
678
679 assert_param(pxConditionWheel != NULL);
680
681 // Find ptr of present Hall sensor ucCondition
682 for(findPtr = 0; findPtr < COUNT_VALID_HALL_COMBINATION; findPtr++)
683 {
684 if (pxConditionWheel->xCondHallSensor.peSequency[findPtr] == ucCondition) break;
685 }
686 // Check for valid present ucCondition of Hall sensor
687 if (findPtr == COUNT_VALID_HALL_COMBINATION)
688 {
689 ret = NO_VALID_HALL_CONDITION;
690 goto return_prvGotoValidHallCondition;
691 }
692
693 // Calculation of the number of steps by sensor
694 if (findPtr >= pxConditionWheel->xCondHallSensor.ucPtr)
695 {
696 countStep = findPtr - pxConditionWheel->xCondHallSensor.ucPtr;
697 if (countStep > (COUNT_VALID_HALL_COMBINATION / 2))
698 countStep = countStep - COUNT_VALID_HALL_COMBINATION;
699 }
700 if (findPtr < pxConditionWheel->xCondHallSensor.ucPtr)
701 {
702 countStep = findPtr - pxConditionWheel->xCondHallSensor.ucPtr;
703 if (countStep < (-COUNT_VALID_HALL_COMBINATION / 2) )
704 countStep = COUNT_VALID_HALL_COMBINATION + countStep;
705 }
706
707 // Remember new ucCondition
708 xSemaphoreTake(pxConditionWheel->xMutexThis, portMAX_DELAY);
709 pxConditionWheel->xCondHallSensor.ucPtr = findPtr;
710 xSemaphoreGive(pxConditionWheel->xMutexThis);
711
712 if (pxConditionWheel->eIdWheel == ID_RIGHT_WHEEL)
713 countStep = countStep * (-1);
714 ret = countStep;
715
716return_prvGotoValidHallCondition:
717 return ret;
718}

1298
1299 // Calculate CCR values
1300 float fRealDutyCycle = prvCalculationRealDutyCycle(pxConditionWheel, fDutyCycle);
1301 uint32_t tempCurrCCR = (uint32_t)( (float)(fabsf(fDutyCycle) / 100.0) *
1302 (float)((float)pxConditionWheel->xAddSetting.periodPwm01Mcs * SCALE_ARR_MCS) );

1305 uint32_t ulTempCcr1, ulTempCcr2, ulTempCcr3;
1306 if (fRealDutyCycle >= EPSILON)

1313 ulTempCcr2 = prvCalculationSinusCcr(pxConditionWheel, tempRealCCR, pxConditionWheel->xSinusControlData.pucSequencyTableBack_B);
1314 ulTempCcr3 = prvCalculationSinusCcr(pxConditionWheel, tempRealCCR, pxConditionWheel->xSinusControlData.pucSequencyTableBack_C);

1318 taskENTER_CRITICAL();
1319 pxConditionWheel->pxHtimPwm->Instance->CCR1 = ulTempCcr1;
1320 pxConditionWheel->pxHtimPwm->Instance->CCR2 = ulTempCcr2;
1321 pxConditionWheel->pxHtimPwm->Instance->CCR3 = ulTempCcr3;
1322 taskEXIT_CRITICAL();
1323
1324 // Update pxConditionWheel data
1325 if ((SCB->ICSR & 0x000001ff) == 0x00)
1326 xSemaphoreTake(pxConditionWheel->xMutexThis, portMAX_DELAY);

1328 pxConditionWheel->ulValueCCR = tempCurrCCR;
1329 pxConditionWheel->xSoftStartStop.fRealSoftDutyCycle = fRealDutyCycle;
1330 pxConditionWheel->xSoftStartStop.ulRealSoftValueCCR = tempRealCCR;
1331 if ((SCB->ICSR & 0x000001ff) == 0x00)
1332 xSemaphoreGive(pxConditionWheel->xMutexThis);
1334}
