247void SetDutyCycle(TConditionWheelPhyLevel *condition, float dutyCycle)
248{
249 assert_param(condition != NULL);
250
251 if (dutyCycle > 100.0){
252 dutyCycle = 100.0;
253 }
254 if (dutyCycle < -100.0){
255 dutyCycle = -100.0;
256 }
257 // Soft change of duty cycle
258 float diff = dutyCycle - condition->xSoftStartStop.fRealSoftDutyCycle;
259 if ((diff > EPSILON) && (diff > condition->xSoftStartStop.fStepSoftDutyCycle))
260 {
261 diff = condition->xSoftStartStop.fStepSoftDutyCycle;
262 }
263 if ((diff < EPSILON) && (fabsf(diff) > condition->xSoftStartStop.fStepSoftDutyCycle))
264 {
265 diff = -(condition->xSoftStartStop.fStepSoftDutyCycle);
266 }
267 float realDutyCycle;
268 if (systemTime_10mcsec < condition->xSoftStartStop.ullPrevTimeChangePWM_10mcs) {condition->xSoftStartStop.ullPrevTimeChangePWM_10mcs = systemTime_10mcsec;}
269 if (systemTime_10mcsec - condition->xSoftStartStop.ullPrevTimeChangePWM_10mcs > condition->xSoftStartStop.ulPeriodSoftChange_10mcs)
270 {
271 realDutyCycle = condition->xSoftStartStop.fRealSoftDutyCycle + diff;
272 condition->xSoftStartStop.ullPrevTimeChangePWM_10mcs = systemTime_10mcsec;
273 } else {
274 realDutyCycle = condition->xSoftStartStop.fRealSoftDutyCycle;
275 }
276
277 uint32_t tempCurrCCR = (uint32_t)((float)(fabsf(dutyCycle)/100.0) * (float)((float)condition->addSetting.periodPwm01Mcs*SCALE_ARR_MCS));
278 uint32_t tempRealCCR = (uint32_t)((float)(fabsf(realDutyCycle)/100.0) * (float)((float)condition->addSetting.periodPwm01Mcs*SCALE_ARR_MCS));
279
280 // Setting duty cycle to the registers
281 if (condition->pHtimPwm->Instance->CCR1 != VALUE_CCR_MAX)
282 {
283 condition->pHtimPwm->Instance->CCR1 = tempRealCCR;
284 }
285 if (condition->pHtimPwm->Instance->CCR2 != VALUE_CCR_MAX)
286 {
287 condition->pHtimPwm->Instance->CCR2 = tempRealCCR;
288 }
289 if (condition->pHtimPwm->Instance->CCR3 != VALUE_CCR_MAX)
290 {
291 condition->pHtimPwm->Instance->CCR3 = tempRealCCR;
292 }
293
294// if ((SCB->ICSR & 0x000001ff) == 0x00)
295// {
296 xSemaphoreTake(condition->xMutexThis, portMAX_DELAY);
297// }
298
299 condition->currDutyCycle = dutyCycle;
300 condition->valueCCR = tempCurrCCR;
301 condition->xSoftStartStop.fRealSoftDutyCycle = realDutyCycle;
302 condition->xSoftStartStop.ulRealSoftValueCCR = tempRealCCR;
303// if ((SCB->ICSR & 0x000001ff) == 0x00)

1152void calcAmplitudeCoef(bool *const isInHysteresis, const TConditionWheelPhyLevel *const condWheel, float *const arrMax, float *const arrMin,
1153 TEnumDirection *const direction, uint8_t *const noForward, uint8_t *const noBack, float *const calibrCoefForward, float *const calibrCoefBack,
1154 float valueAmplitude)
1155{
1156 static const float ANGULAR_HYSTERESIS_RAD = 1.f;
1157
1158 // Если внутри гистерезиса - ждём, когда выйдем оттуда и начнём поиск нового экстремума
1159 if (*isInHysteresis == true)
1160 {
1161 if (condWheel->currentAngular > ANGULAR_HYSTERESIS_RAD)
1162 {
1163 *isInHysteresis = false; // вышли из петли гистерезиса
1164 arrMax[*noForward] = 0.f;
1165 *direction = FORWARD;
1166 }
1167 if (condWheel->currentAngular < -(ANGULAR_HYSTERESIS_RAD))
1168 {
1169 *isInHysteresis = false; // вышли из петли гистерезиса
1170 arrMin[*noBack] = 0.f;
1171 *direction = BACK;
1172 }
1173 // Если вне гистерезиса - ищем экстремум; и при заходе в гистерезис - запоминаем значение экстремума и если надо подсчитываем новые коэффициенты
1174 } else {
1175 if (*direction == FORWARD)
1176 {
1177 if (condWheel->currentAngular > arrMax[*noForward]) {
1178 arrMax[*noForward] = condWheel->currentAngular;
1179 }
1180 if (fabsf(condWheel->currentAngular) < EPSILON_ANGULAR)
1181 {
1182 *isInHysteresis = true;
1183 (*noForward)++;
1184 if (*noForward >= COUNT_AMPLITUDE_VALUE_CALIBR)
1185 {
1186 float fSum = 0.f, fAvg;
1187 for(uint8_t i=0;i<COUNT_AMPLITUDE_VALUE_CALIBR;i++)
1188 {
1189 fSum = fSum + arrMax[i];
1190 }
1191 fAvg = fSum / (float)(COUNT_AMPLITUDE_VALUE_CALIBR);
1192 *calibrCoefForward = *calibrCoefForward * fabsf(valueAmplitude / fAvg);
1193 if (uCountCycleNotAutocalibr < COUNT_CYCLES_NOT_AUTOCALIBRATION)
1194 {
1195 uCountCycleNotAutocalibr++;
1196 *calibrCoefForward = 1.0;

1203 }
1204 }

1206 }

1216 (*noBack)++;

1218 {
