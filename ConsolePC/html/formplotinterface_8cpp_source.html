161void FormPlotInterface::SlotUpdateDisplay()
162{
163 TTelemetryValuesDisplay *atv = &avgTelemetryValues; // Меньше писать кода, необязательная переменная
164
165 // Количество ошибок всегда обновляется
166 ui->lErrorCrc->setText(("Ошибки Crc: " + QString::number(avgTelemetryValues.crcError)));
167 // Если не было данных, то ничего не делать
168 if (avgKolMeasure == 0)
169 {
170 atv->valueW1 = atv->valueW2 = atv->valueQ1 = atv->valueQ2 = atv->valueS1 = atv->valueS2 = 0;
171 atv->errHall1 = atv->errHall2 = 0;
172 atv->systemTime = atv->crcError = 0;
173 goto return_SlotUpdateDisplay;
174 }
175
176 // Заполнение форм
177 ui->leQ_left->setText(QString::number( (atv->valueQ1 / avgKolMeasure) ));
178 ui->leQ_right->setText(QString::number( (atv->valueQ2 / avgKolMeasure) ));
179 ui->leW_left->setText(QString::number( (atv->valueW1 / avgKolMeasure) ));
180 ui->leW_right->setText(QString::number( (atv->valueW2 / avgKolMeasure) ));
181 ui->leDutyCycle1->setText(QString::number( (atv->valueS1 / avgKolMeasure) ));
182 ui->leDutyCycle2->setText(QString::number( (atv->valueS2 / avgKolMeasure) ));
183 ui->leErroHal1->setText(QString::number( (atv->errHall1) ));
184 ui->leErroHal2->setText(QString::number( (atv->errHall2) ));
185 ui->leDifferenceW->setText(QString::number( ((atv->valueW1 - atv->valueW2) / avgKolMeasure) ));
186 ui->lePhaseShift->setText(QString::number( ((atv->valueQ1 - atv->valueQ2) / avgKolMeasure) ));
187
188 if (terminalMessage->isEmpty() == false)
189 {
190 this->ui->pteTerminal->appendPlainText(*terminalMessage);
191 terminalMessage->clear();
192 }
193
194 // Сброс осреднённых данных
195 memset(&avgTelemetryValues, 0, sizeof(avgTelemetryValues)); // Сброс осреднённых данных
196 avgKolMeasure = 0;
197
198 ui->plotWidget->replot();
199
200return_SlotUpdateDisplay:
201 return;

226void FormPlotInterface::on_RecieveDataTelemetry(TTelemetryValuesDisplay telemetry)
227{
229 qreal timeSec = (qreal)telemetry.systemTime * SCALING_FACTORS.SCALE_TIME;
230 qreal valueQ1 = telemetry.valueQ1 * SCALING_FACTORS.SCALE_Q;
231 qreal valueQ2 = telemetry.valueQ2 * SCALING_FACTORS.SCALE_Q;
232 qreal valueW1 = telemetry.valueW1 * SCALING_FACTORS.SCALE_W;
233 qreal valueW2 = telemetry.valueW2 * SCALING_FACTORS.SCALE_W;
234 qreal valueS1 = telemetry.valueS1 * SCALING_FACTORS.SCALE_S;
235 qreal valueS2 = telemetry.valueS2 * SCALING_FACTORS.SCALE_S;
236
237 // Отображение на графиках
238 ui->plotWidget->graph(0)->addData(timeSec, valueQ1);
239 ui->plotWidget->graph(1)->addData(timeSec, valueQ2);
240 ui->plotWidget->graph(2)->addData(timeSec, valueW1);
241 ui->plotWidget->graph(3)->addData(timeSec, valueW2);
242
243 // Запись в файл
244 if (modeRecordOn == true)
245 {
246 fileLog.write(QString::number(valueQ1).toUtf8() + "\t" + QString::number(valueQ2).toUtf8() + "\t" +
247 QString::number(valueW1).toUtf8() + "\t" + QString::number(valueW2).toUtf8() + "\t" +
248 QString::number(valueS1).toUtf8() + "\t" + QString::number(valueS2).toUtf8() + "\t" +
249 QString::number(valueW1-valueW2).toUtf8() + "\t" + QString::number(valueQ1-valueQ2).toUtf8() + "\t" +
250 QString::number(telemetry.errHall1).toUtf8() + "\t" + QString::number(telemetry.errHall2).toUtf8() + "\t" +
251 QString::number(telemetry.crcError).toUtf8() + "\t" + QString::number(timeSec).toUtf8() + "\n");
252 }
253
254 // Осреднение данных
255 this->avgKolMeasure++;
256 this->avgTelemetryValues.valueQ1 = this->avgTelemetryValues.valueQ1 + valueQ1;
257 this->avgTelemetryValues.valueQ2 = this->avgTelemetryValues.valueQ2 + valueQ2;
258 this->avgTelemetryValues.valueW1 = this->avgTelemetryValues.valueW1 + valueW1;
259 this->avgTelemetryValues.valueW2 = this->avgTelemetryValues.valueW2 + valueW2;
260 this->avgTelemetryValues.valueS1 = this->avgTelemetryValues.valueS1 + valueS1;
261 this->avgTelemetryValues.valueS2 = this->avgTelemetryValues.valueS2 + valueS2;
262 this->avgTelemetryValues.errHall1 = telemetry.errHall1;
263 this->avgTelemetryValues.errHall2 = telemetry.errHall2;
264 this->avgTelemetryValues.crcError = telemetry.crcError;
265}
