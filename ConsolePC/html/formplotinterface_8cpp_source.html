145void FormPlotInterface::SlotUpdateDisplay()
146{
147 // Количество ошибок всегда обновляется
148 ui->lErrorCrc->setText(("Ошибки Crc: " + QString::number(avgTelemetryValues.crcError)));
149 // Если не было данных, то ничего не делать
150 if(avgKolMeasure == 0)
151 {
152#define atv avgTelemetryValues
153 atv.valueW1 = atv.valueW2 = atv.valueQ1 = atv.valueQ2 = atv.valueS1 = atv.valueS2 = 0;
154 atv.errHall1 = atv.errHall2 = 0;
155 atv.systemTime = atv.crcError = 0;
156#undef atv
157 return;
158 }
159 // Заполнение форм
160 ui->leQ_left->setText(QString::number( (avgTelemetryValues.valueQ1 / avgKolMeasure) ));
161 ui->leQ_right->setText(QString::number( (avgTelemetryValues.valueQ2 / avgKolMeasure) ));
162 ui->leW_left->setText(QString::number( (avgTelemetryValues.valueW1 / avgKolMeasure) ));
163 ui->leW_right->setText(QString::number( (avgTelemetryValues.valueW2 / avgKolMeasure) ));
164 ui->leDutyCycle1->setText(QString::number( (avgTelemetryValues.valueS1 / avgKolMeasure) ));
165 ui->leDutyCycle2->setText(QString::number( (avgTelemetryValues.valueS2 / avgKolMeasure) ));
166 ui->leErroHal1->setText(QString::number( (avgTelemetryValues.errHall1) ));
167 ui->leErroHal2->setText(QString::number( (avgTelemetryValues.errHall2) ));
168 ui->leDifferenceW->setText(QString::number( ((avgTelemetryValues.valueW1-avgTelemetryValues.valueW2) / avgKolMeasure) ));
169 ui->lePhaseShift->setText(QString::number( ((avgTelemetryValues.valueQ1-avgTelemetryValues.valueQ2) / avgKolMeasure) ));
170 if (terminalMessage->isEmpty() == false)
171 {
172 this->ui->pteTerminal->appendPlainText(*terminalMessage);
173 terminalMessage->clear();
174 }
175
176 // Сброс осреднённых данных
177 memset(&avgTelemetryValues, 0, sizeof(avgTelemetryValues)); // Сброс осреднённых данных
178 avgKolMeasure = 0;
179
180 ui->plotWidget->replot();
181}

208void FormPlotInterface::on_RecieveDataTelemetry(TTelemetryValuesDisplay telemetry)
209{
211 qreal timeSec = (qreal)telemetry.systemTime * SCALING_FACTORS.SCALE_TIME;
212 qreal valueQ1 = telemetry.valueQ1 * SCALING_FACTORS.SCALE_Q;
213 qreal valueQ2 = telemetry.valueQ2 * SCALING_FACTORS.SCALE_Q;
214 qreal valueW1 = telemetry.valueW1 * SCALING_FACTORS.SCALE_W;
215 qreal valueW2 = telemetry.valueW2 * SCALING_FACTORS.SCALE_W;
216 qreal valueS1 = telemetry.valueS1 * SCALING_FACTORS.SCALE_S;
217 qreal valueS2 = telemetry.valueS2 * SCALING_FACTORS.SCALE_S;
218
219 // Отображение на графиках
220 ui->plotWidget->graph(0)->addData(timeSec, valueQ1);
221 ui->plotWidget->graph(1)->addData(timeSec, valueQ2);
222 ui->plotWidget->graph(2)->addData(timeSec, valueW1);
223 ui->plotWidget->graph(3)->addData(timeSec, valueW2);
224
225 // Запись в файл
226 if(modeRecordOn == true)
227 {
228 fileLog.write(QString::number(valueQ1).toUtf8() + "\t" + QString::number(valueQ2).toUtf8() + "\t" +
229 QString::number(valueW1).toUtf8() + "\t" + QString::number(valueW2).toUtf8() + "\t" +
230 QString::number(valueS1).toUtf8() + "\t" + QString::number(valueS2).toUtf8() + "\t" +
231 QString::number(valueW1-valueW2).toUtf8() + "\t" + QString::number(valueQ1-valueQ2).toUtf8() + "\t" +
232 QString::number(telemetry.errHall1).toUtf8() + "\t" + QString::number(telemetry.errHall2).toUtf8() + "\t" +
233 QString::number(telemetry.crcError).toUtf8() + "\t" + QString::number(timeSec).toUtf8() + "\n");
234 }
235 // Осреднение данных
236 this->avgKolMeasure++;
237 this->avgTelemetryValues.valueQ1 = this->avgTelemetryValues.valueQ1 + valueQ1; this->avgTelemetryValues.valueQ2 = this->avgTelemetryValues.valueQ2 + valueQ2;
238 this->avgTelemetryValues.valueW1 = this->avgTelemetryValues.valueW1 + valueW1; this->avgTelemetryValues.valueW2 = this->avgTelemetryValues.valueW2 + valueW2;
239 this->avgTelemetryValues.valueS1 = this->avgTelemetryValues.valueS1 + valueS1; this->avgTelemetryValues.valueS2 = this->avgTelemetryValues.valueS2 + valueS2;
240 this->avgTelemetryValues.errHall1 = telemetry.errHall1; this->avgTelemetryValues.errHall2 = telemetry.errHall2;
241 this->avgTelemetryValues.crcError = telemetry.crcError;
242}
