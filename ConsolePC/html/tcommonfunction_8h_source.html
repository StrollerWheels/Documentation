92 template<typename TPacketStatus, typename TPacketData>
93 static bool RecieveComPort (QSerialPort *comPort, int32_t timeMs, TPacketData *packetData, uint8_t packetDataId, QString *strResult)
94 {
95 unsigned char bufIn[MAX_SIZE_PACKET*100] = {0}; // Пакет с ответом от Блока управления
96 unsigned char bufInPrepared[MAX_SIZE_PACKET] = {0}; // Пакет с ответом от Блока управления, подготовленный
97 QTime timeReception; // Время принятия данных
98 uint16_t lenReadBuf = 0, nBufIn, nBufPrepared = 0;
99 TPacketStatus *pStatus = reinterpret_cast<TPacketStatus*>(bufInPrepared);
100 TPacketData *pPacketData = reinterpret_cast<TPacketData*>(bufInPrepared);
101 bool ret = true;
102
103
104 QThread::msleep(TIME_SLEEP_MS);
105
106 timeReception.start();
107 // Чтение данных в течении timeMs
108 while (timeReception.elapsed() < timeMs)
109 {
110 if (comPort->waitForReadyRead(PERIOD_POLL_MS) == true)
111 {
112 nBufIn=0; // Обнуление номера принятых байт в данной итерации
113 lenReadBuf = comPort->read(reinterpret_cast<char*>(bufIn), MAX_SIZE_PACKET * 100);
114
115 if (nBufPrepared == 0)
116 {
117 // Поиск преамбулы начала кадра 0хАААА
118 for(nBufIn = 0; nBufIn < lenReadBuf; nBufIn++)
119 {
120 if ((bufIn[nBufIn] == 0xAA) && (bufIn[nBufIn + 1] == 0xAA) &&
121 ((bufIn[nBufIn + 2] == TYPE_PACKET.ANSWER_STATUS) || (bufIn[nBufIn + 2] == packetDataId)))
122 {
123 // Когда дошли до преамбулы пакета 0хАААА
124 break;
125 }
126 }
127 }
128
129 for ( ; nBufIn < lenReadBuf; nBufIn++)
130 {
131 bufInPrepared[nBufPrepared] = bufIn[nBufIn];
132 nBufPrepared++;
133 }
134
135 // Если пакет уже принят, то выходим из цикла ожидания while
136 if ((bufInPrepared[2] == TYPE_PACKET.ANSWER_STATUS) && (nBufPrepared >= sizeof(TPacketStatus)))
137 {
138 lenReadBuf = sizeof(TPacketStatus);
139 break;
140 }
141
142 if ((bufInPrepared[2] == packetDataId) && (nBufPrepared >= sizeof(TPacketData)))
143 {
144 lenReadBuf = sizeof(TPacketData);
145 break;
146 }
147 }
148 }
149
150 // Проверка корректности длины принятого пакета
151 if ((lenReadBuf != sizeof(TPacketStatus)) && (lenReadBuf != sizeof(TPacketData)))
152 {
153 *strResult = "Неверный ответ Блока управления";
154 ret = false;
155 goto return_RecieveComPort;
156 }
157
158 // Если пришёл пакет Статус выполнения команды
159 if (lenReadBuf == sizeof(TPacketStatus))
160 {
161 if ((pStatus->idPacket != TYPE_PACKET.ANSWER_STATUS) ||
162 (pStatus->crc16 != crc16citt(bufInPrepared,sizeof(TPacketStatus) - 2)) ||
163 (STATUS_ANSWER_MAP.count(pStatus->status) == 0) )
164 {
165 *strResult = "Неверный ответ Блока управления";
166 ret = false;
167 goto return_RecieveComPort;
168 }
169
170 *strResult = QString::fromUtf8(STATUS_ANSWER_MAP[pStatus->status].c_str());
171 }
172
173 // Если пришёл пакет TPacketData
174 if ((lenReadBuf == sizeof(TPacketData)) &&
175 (pPacketData->crc16 != crc16citt(bufInPrepared,sizeof(TPacketData) - 2)))
176 {
177 *strResult = "Неверный ответ Блока управления";
178 ret = false;
179 goto return_RecieveComPort;
180 }
181
182 // Запомнить целый и правильный пакет
183 *strResult = "Команда выполнена успешно";
184 *packetData = *pPacketData;
185
186return_RecieveComPort:
187 return ret;
188 }
