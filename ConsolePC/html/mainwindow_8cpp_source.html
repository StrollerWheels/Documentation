305void MainWindow::on_actionReadRAM_triggered()
306{
307 TProtocolPacketCommandReal packetCmd; // Пакет с командой на чтение
308 QSerialPort comPort; // Локальный сом порт
309 TProtocolPacketSettings packetSettingsIn; // Принятые пакет с настройками
310 QSerialPort comPortCmd; // Локальный сом порт
311 QString strResult; // Результат в виде сообщения выполнения общих функций класса TCommonFunction
312
313 // Формирование пакета
314 packetCmd.preambule = PREAMBULE_PACKET;
315 packetCmd.idPacket = TYPE_PACKET.CMD_READ_SETTINGS_FROM_RAM;
316 packetCmd.crc16 = crc16citt((unsigned char*)(&packetCmd), (sizeof(packetCmd) - 2));
317
318 // Открытие СОМ порта
319 if (TCommonFunction::OpenComPort(&comPortCmd,this->comPortSettings, &strResult) == false)
320 {
321 QMessageBox::warning(this, "Предупреждение", strResult);
322 goto return_on_actionReadRAM_triggered;
323 }
324
325 // Послыка пакета с настройками
326 if (TCommonFunction::SendComPort<TProtocolPacketCommandReal> (&comPortCmd,packetCmd,&strResult) == false)
327 {
328 QMessageBox::warning(this, "Предупреждение", strResult);
329 comPortCmd.close();
330 goto return_on_actionReadRAM_triggered;
331 }
332
333 // Принятие пакета в ответ
334 if (TCommonFunction::RecieveComPort<TProtocolPacketStatusAnswer, TProtocolPacketSettings>
335 (&comPortCmd, 2000, &packetSettingsIn, TYPE_PACKET.ANSWER_READ_SETTINGS_FROM_RAM, &strResult) == false)
336 {
337 QMessageBox::warning(this, "Предупреждение", strResult);
338 comPortCmd.close();
339 goto return_on_actionReadRAM_triggered;
340 }
341
342 comPortCmd.close();
343 QMessageBox::information(this, "Результат команды", strResult);
344
345 // Отображение принятых настроек управеления
346 this->DisplayAcceptSettings(packetSettingsIn);
347
348return_on_actionReadRAM_triggered:
349 return;
350}
