23const uint8_t NUMBER_OF_ATTEMPT (2);
24bool TCommonFunction::OpenComPort(QSerialPort *comPort, TComSettings settings, QString *strResult)
25{
26 uint8_t attemptNumber = 0;
27
28 // Проверка корректности заполнения настроек COM порта
29 if(settings.portName.isEmpty())
30 {
31 *strResult = "Выберите настройки СОМ порта";
32 return false;
33 }
34 // Настройка СОМ порта
35 comPort->setPortName(settings.portName);
36 comPort->setBaudRate(settings.speedRate);
37 comPort->setDataBits(QSerialPort::Data8);
38 comPort->setFlowControl(QSerialPort::NoFlowControl);
39 comPort->setParity(QSerialPort::NoParity);
40 comPort->setStopBits(QSerialPort::OneStop);
41 while ((attemptNumber < NUMBER_OF_ATTEMPT) && (comPort->isOpen()==false))
42 {
43 comPort->close();
44 if(comPort->open(QSerialPort::ReadWrite)==false)
45 {
46 attemptNumber++;
47 } else {
48 break;
49 }
50 }
51 if (attemptNumber >= NUMBER_OF_ATTEMPT)
52 {
53 *strResult = "COM порт не может быть открыт";
54 return false;
55 }
56 // СОМ порт открыт
57 *strResult = "";
58 return true;
59}
