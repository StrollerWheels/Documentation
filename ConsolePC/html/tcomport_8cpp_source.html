149

164 emit SignalStatusAnswer(strResult);
165 timerRead->start(PERIOD_TIMER_READ_MS);
166 return;
167 }
168 // Приянтие пакета в ответ
169 if( this->RecieveComPort<TProtocolPacketStatusAnswer, TProtocolPacketStatusAnswer> (PERIOD_TIMER_RECIEVE_MS, TYPE_PACKET.ANSWER_STATUS, &packetAnswer,&strResult) == false)
170 {
171 emit SignalStatusAnswer(strResult);
172 timerRead->start(PERIOD_TIMER_READ_MS);
173 return;
174 }
175 // Команда выполнена успешно ДО СЮДА НЕ ДОЙДЁТ ПОКА

285 packet.preambule = PREAMBULE_PACKET;
