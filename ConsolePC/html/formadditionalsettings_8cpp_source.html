68void FormAdditionalSettings::on_pbWriteSettings_clicked()
69{
70 TProtocolPacketAdditionalSettings packetOut; // Пакет с настройками для передачи
71 TProtocolPacketStatusAnswer packetAnswer; // Пакет с ответом
72 QSerialPort comPort; // Локальный сом порт
73 QString strResult; // Результат в виде сообщения выполнения общих функций класса TCommonFunction
74
75 // Проверка корректности заполнения настроек
76 if ((ui->lePeriodPwm->text().isEmpty()) || (ui->leTimeBounce->text().isEmpty()) ||
77 (ui->leMaxAmplitude->text().isEmpty()) || (ui->leMaxDifferenceBetweenAngles->text().isEmpty()) ||
78 (ui->leMaxLinearVelocity->text().isEmpty()))
79 {
80 QMessageBox::warning(this,"Предупреждение", "Не заданы настройки");
81 goto return_on_pbWriteSettings_clicked;
82 }
83
84 // Запомнили настройки из интерфейса
85 this->additionalSetttings.periodPwm01Mcs = ui->lePeriodPwm->text().toInt();
86 this->additionalSetttings.timeBounce10Mcs = ui->leTimeBounce->text().toInt();
87 this->additionalSetttings.typeControl = static_cast<TEnumTypeControl>(ui->cbTypeControl->currentIndex());
88 this->additionalSetttings.freqTelemetry = ui->leFreqTelemetry->text().toInt();
89 this->additionalSetttings.maxAmplitude = ui->leMaxAmplitude->text().toFloat();
90 this->additionalSetttings.maxDiffAngles = ui->leMaxDifferenceBetweenAngles->text().toFloat();
91 this->additionalSetttings.maxLinearVelosity = ui->leMaxLinearVelocity->text().toFloat();
92
93 // Формирование пакета
94 packetOut.preambule = PREAMBULE_PACKET;
95 packetOut.idPacket = TYPE_PACKET.CMD_WRITE_ADDITIONAL_SETTINGS;
96 packetOut.setting = this->additionalSetttings;
97 packetOut.crc16 = crc16citt((unsigned char*)(&packetOut), (sizeof(packetOut) - 2));
98
99 // Открытие СОМ порта
100 if (TCommonFunction::OpenComPort(&comPort, *(this->comSettings), &strResult) == false)
101 {
102 QMessageBox::warning(this,"Предупреждение", strResult);
103 goto return_on_pbWriteSettings_clicked;
104 }
105
106 // Послыка пакета с настройками
107 if (TCommonFunction::SendComPort<TProtocolPacketAdditionalSettings>(&comPort,packetOut, &strResult) == false)
108 {
109 QMessageBox::warning(this,"Предупреждение", strResult);
110 comPort.close();
111 goto return_on_pbWriteSettings_clicked;
112 }
113
114 // Принятие пакета в ответ
115 if (TCommonFunction::RecieveComPort<TProtocolPacketStatusAnswer, TProtocolPacketStatusAnswer>
116 (&comPort, 500, &packetAnswer, TYPE_PACKET.ANSWER_STATUS, &strResult) == false)
117 {
118 QMessageBox::warning(this,"Предупреждение", strResult);
119 comPort.close();
120 goto return_on_pbWriteSettings_clicked;
121 }
122
123 // Команда выполнена успешно
124 QMessageBox::information(this,"Результат команды", strResult);
125
126return_on_pbWriteSettings_clicked:
127 comPort.close();
128}

135{
136 TProtocolPacketCommandReal packetCmd; // Пакет с командой на чтение
137 QSerialPort comPort; // Локальный сом порт

142 packetCmd.preambule = PREAMBULE_PACKET;
143 packetCmd.idPacket = TYPE_PACKET.CMD_READ_ADDITIONAL_SETTINGS;
144 packetCmd.crc16 = crc16citt(reinterpret_cast<unsigned char*>(&packetCmd), (sizeof(packetCmd) - 2));

147 if (this->comSettings->portName.isEmpty())

149 QMessageBox::warning(this, "Предупреждение", "Выберите настройки СОМ порта");

157 goto return_on_pbReadSettings_clicked;
158 }
159
160 // Послыка пакета с настройками
161 if (TCommonFunction::SendComPort<TProtocolPacketCommandReal> (&comPort,packetCmd, &strResult) == false)
162 {
163 QMessageBox::warning(this, "Предупреждение", strResult);
164 comPort.close();
165 goto return_on_pbReadSettings_clicked;
166 }
167
168 // Принятие пакета в ответ
169 if (TCommonFunction::RecieveComPort<TProtocolPacketStatusAnswer, TProtocolPacketAdditionalSettings>
170 (&comPort, 1000, &packetAnswer, TYPE_PACKET.ANSWER_READ_ADDITIONAL_SETTINGS, &strResult) == false)
171 {
172 QMessageBox::warning(this,"Предупреждение", strResult);
173 comPort.close();
174 goto return_on_pbReadSettings_clicked;
175 }
176
177 // Запомнить пришедшие настройки
178 this->additionalSetttings = packetAnswer.setting;
179 this->ui->lePeriodPwm->setText(QString::number(additionalSetttings.periodPwm01Mcs));
180 this->ui->leTimeBounce->setText(QString::number(additionalSetttings.timeBounce10Mcs));
181 this->ui->leFreqTelemetry->setText(QString::number(additionalSetttings.freqTelemetry));
182 this->ui->cbTypeControl->setCurrentIndex(static_cast<int>(additionalSetttings.typeControl));
183 this->ui->leMaxAmplitude->setText(QString::number(additionalSetttings.maxAmplitude));
184 this->ui->leMaxDifferenceBetweenAngles->setText(QString::number(additionalSetttings.maxDiffAngles));
185 this->ui->leMaxLinearVelocity->setText(QString::number(additionalSetttings.maxLinearVelosity));
186 // Команда выполнена успешно
187 QMessageBox::information(this,"Результат команды", strResult);
188
189return_on_pbReadSettings_clicked:
