150TEnumApiControlResult xMyFunctionControl (bool bChangeRamCoef)
151{
152 const float PING_TIME_INTERVAL = 5.f;
153
154 TApiControlEvent pxEvents[NUMBER_OF_EVENTS_STORED]; // Array with events
155 TEnumApiControlResult eResult = API_RESULT_OK;
156 uint8_t ucKolEvent;
157 float fRealTime; // real time
158 float fRealLeftW = 0, fRealRightW = 0;
159 float fFreqCall = 30.f;
160
161 static TControlEventQT xEventLeftPrev_ = {.fQ = 0.f, .fTime = 0.f}, xEventRightPrev_ = {.fQ = 0.f, .fTime = 0.f}; // value in beginning measure
162 static TControlEventQT xEventLeftCurr_ = {.fQ = 0.f, .fTime = 0.f}, fEventRightCurr_ = {.fQ = 0.f, .fTime = 0.f}; // current value
163 static TApiControlCoef xCurrCoef_ = {0}; // Control regulator coefficients
164 static float fOutLeftS_ = 0, fOutRightS_ = 0; // Duty cycle
165 static TControlError xErrorLeft_ = {0}, xErrorRight_ = {0}; // PID error regulator
166
167 // Get coefficients from RAM setting if need
168 if (bChangeRamCoef == true)
169 {
170 pxApiTacControlFunc->bGetCoefficients(&xCurrCoef_);
171 memset(&xEventLeftPrev_, 0, sizeof(xEventLeftPrev_));
172 memset(&xEventRightPrev_, 0, sizeof(xEventRightPrev_));
173 memset(&xEventLeftCurr_, 0, sizeof(xEventLeftCurr_));
174 memset(&fEventRightCurr_, 0, sizeof(fEventRightCurr_));
175 memset(&xErrorLeft_, 0, sizeof(xErrorLeft_));
176 memset(&xErrorRight_, 0, sizeof(xErrorRight_));
177 memset(&xEventLeftQT_, 0, sizeof(xEventLeftQT_));
178 memset(&xEventRightQT_, 0, sizeof(xEventRightQT_));
179 }
180
181 // Reading data
182 fRealTime = SCALE_TIME_API * (float)pxApiTacControlFunc->ullGetSystemTime_10Mcs();
183 ucKolEvent = pxApiTacControlFunc->ulGetLastEvents (pxEvents, NUMBER_OF_EVENTS_STORED);
184 fFreqCall = (float)pxApiTacControlFunc->ulGetFrequencyFunctionCall();
185
186 // Instead of pinging
187 static float prevTime_ = 0.f;
188 if (fRealTime - prevTime_ > PING_TIME_INTERVAL)
189 {
190 prevTime_ = fRealTime;
191 pxApiTacControlFunc->eApiPrintf("Hello");
192 }
193
194 // Remember last event and average W and T
202 for (uint8_t i = 0; i < ucKolEvent; i++)
203 {
204 if (pxEvents[i].eWheel == LEFT_WHEEL_API_CONTROL)
205 {
206 xEventLeftCurr_.fQ = pxEvents[i].fAngular;
207 xEventLeftCurr_.fTime = SCALE_TIME_API * (float)pxEvents[i].ullTime;
208 prvPushEventQT(xEventLeftCurr_.fQ, xEventLeftCurr_.fTime, &xEventLeftQT_);
209 xEventLeftPrev_ = xEventLeftCurr_;
210 }
211 if (pxEvents[i].eWheel == RIGHT_WHEEL_API_CONTROL)
212 {
213 fEventRightCurr_.fQ = pxEvents[i].fAngular;
214 fEventRightCurr_.fTime = SCALE_TIME_API * (float)pxEvents[i].ullTime;
215 prvPushEventQT(fEventRightCurr_.fQ, fEventRightCurr_.fTime, &xEventRightQT_);
216 xEventRightPrev_ = fEventRightCurr_;
217 }
218 }
219
220 // CALCULATION ALL TELEMETRY
221 // Calculation angular velocity
222 fRealLeftW = prvCalculationAngularVelocity(LEFT_WHEEL_API_CONTROL, fRealTime);
223 fRealRightW = prvCalculationAngularVelocity(RIGHT_WHEEL_API_CONTROL, fRealTime);
224
225 // Calculation real Q of wheels
226 TControlEventQT xLastEvent[KOL_REMEMBER_QT_EVENT] = {0};
227 prvPopEventQT(xLastEvent, KOL_REMEMBER_QT_EVENT, &xEventLeftQT_, false);
228 float fRealLeftQ = prvCalculationAngular(fRealTime, xLastEvent[0], fRealLeftW);
229 prvPopEventQT(xLastEvent, KOL_REMEMBER_QT_EVENT, &xEventRightQT_, false);
230 float fRealRightQ = prvCalculationAngular(fRealTime, xLastEvent[0], fRealRightW);
231
232 // CALCULATION OF CONTROL PARAMETERS
233 // Calculation duty cycle
234 TApiControlTargetValue xTarget = pxApiTacControlFunc->xGetTargetValue(fRealTime);
235
236 // Left wheel
237 if (xTarget.eLeft == TARGET_Q)
238 fOutLeftS_ = prvCalculationDutyCycle(xCurrCoef_.xPidLeft, &xErrorLeft_, xTarget.fValueLeft, fRealLeftQ, fFreqCall);
239 if (xTarget.eLeft == TARGET_W)
240 fOutLeftS_ = prvCalculationDutyCycle(xCurrCoef_.xPidLeft, &xErrorLeft_, xTarget.fValueLeft, fRealLeftW, fFreqCall);
241 xErrorLeft_.fErrorPrev = xErrorLeft_.fErrorCurr;
242
243 // Right wheel
244 if (xTarget.eRight == TARGET_Q)
245 fOutRightS_ = prvCalculationDutyCycle(xCurrCoef_.xPidRight, &xErrorRight_, xTarget.fValueRight, fRealRightQ, fFreqCall);
246 if (xTarget.eRight == TARGET_W)
247 fOutRightS_ = prvCalculationDutyCycle(xCurrCoef_.xPidRight, &xErrorRight_, xTarget.fValueRight, fRealRightW, fFreqCall);
248 xErrorRight_.fErrorPrev = xErrorRight_.fErrorCurr;
249
250 // Send the calculated data
251 if (fOutLeftS_ > (100.f-EPSILON))
252 fOutLeftS_ = (100.f-EPSILON);
253 if (fOutLeftS_ < (-100.f+EPSILON))
254 fOutLeftS_ = (-100.f+EPSILON);
255 if (fOutRightS_ > (100.f-EPSILON))
256 fOutRightS_ = (100.f-EPSILON);
257 if (fOutRightS_ < (-100.f+EPSILON))
258 fOutRightS_ = (-100.f+EPSILON);
259
260 pxApiTacControlFunc->bSetCoefficients(&xCurrCoef_);
261 pxApiTacControlFunc->bSetVelocityAngular(fRealLeftW, fRealRightW);
262 pxApiTacControlFunc->bSetDutyCycle(fOutLeftS_, fOutRightS_);
263
264 return eResult;
265}

298static uint8_t prvPopEventQT (TControlEventQT *pxEventOut, uint8_t kol, TEventQT *pxStructEv, bool prTransition)
299{
300 int8_t scEventNo = pxStructEv->ucPos;
301 uint8_t i = 0;
302
303 if (kol > KOL_REMEMBER_QT_EVENT)
304 kol = KOL_REMEMBER_QT_EVENT;
305
306 while (i < kol)
307 {
308 pxEventOut[i].fQ = pxStructEv->pxArrayQT[scEventNo].fQ;
309 pxEventOut[i].fTime = pxStructEv->pxArrayQT[scEventNo].fTime;
310 scEventNo--; i++;
311 if (scEventNo < 0)
312 scEventNo = KOL_REMEMBER_QT_EVENT-1;
313 }
314
315 if (prTransition == true)
316 pxStructEv->ucPos = scEventNo;
317
318 return i;
319}

330static float prvCalculationAngularVelocity (TEnumApiControlWheel eIdWheel, float fRealTime)
331{
332 uint8_t ucEventNo = 0;
333 TControlEventQT xLastEvent[KOL_REMEMBER_QT_EVENT] = {0};
334 float dt = 0.f;
335 float ret;
336
337 if (eIdWheel == LEFT_WHEEL_API_CONTROL)
338 {
339 prvPopEventQT(xLastEvent, KOL_REMEMBER_QT_EVENT, &xEventLeftQT_, false);
340 } else {
341 prvPopEventQT(xLastEvent, KOL_REMEMBER_QT_EVENT, &xEventRightQT_, false);
342 }
343
344 dt = xLastEvent[0].fTime - xLastEvent[1].fTime;
345 while ((ucEventNo < KOL_REMEMBER_QT_EVENT-2) && (dt < TIME_TO_CALC_W))
346 {
347 dt = xLastEvent[0].fTime-xLastEvent[ucEventNo+2].fTime;
348 ucEventNo++;
349 }
350 if (ucEventNo == 0)
351 {
352 ret = 0.f;
353 } else {
354 if (dt > TIME_TO_CALC_W)
355 {
356 dt = xLastEvent[0].fTime - xLastEvent[ucEventNo].fTime;
357 ret = (xLastEvent[0].fQ - xLastEvent[ucEventNo].fQ) / dt;
358 } else {
359 ret = (xLastEvent[0].fQ - xLastEvent[ucEventNo+1].fQ) / dt;
360 }
361 }
362 if ((fRealTime - xLastEvent[0].fTime) > TIME_TO_W_0)
363 ret = 0.f;
364
365 return ret;
366}

376static float prvCalculationAngular (float fRealTime, TControlEventQT xLastEvent, float fW)
377{
378 float ret;
379 float dQ = fW * (fRealTime - xLastEvent.fTime);
380
381 if (fabs(dQ) < SENSITIVITY_Q)
382 {
383 ret = xLastEvent.fQ + dQ;
384 } else {
385 ret = xLastEvent.fQ + ((dQ > 0) ? SENSITIVITY_Q:(-SENSITIVITY_Q));
386 }
387
388 return ret;
389}

405static float prvCalculationDutyCycle (TApiControlPidCoef xPidCoef, TControlError *const xError, float fTargetValue,
406 float fRealValue, float fFreq)
407{
408 float ret;
409
410 xError->fErrorCurr = fTargetValue - fRealValue;
411 xError->fErrorIntegral = xError->fErrorIntegral + xError->fErrorCurr;
412
413 ret = xError->fErrorCurr * xPidCoef.fProportional + xError->fErrorIntegral * xPidCoef.fIntegral / fFreq +
414 (xError->fErrorCurr - xError->fErrorPrev) * xPidCoef.fDifferential * fFreq;
415
416 return ret;
417}
