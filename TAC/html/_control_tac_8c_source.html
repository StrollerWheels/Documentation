83uint8_t popEventQT (TControlEventQT *eventOut, uint8_t kol, TEventQT *structEv, bool prTransition)
84{
85 int8_t no = structEv->pos;
86 uint8_t i = 0;
87
88 if (kol > KOL_REMEMBER_QT_EVENT) {
89 kol = KOL_REMEMBER_QT_EVENT;
90 }
91
92 while (i < kol)
93 {
94 eventOut[i].Q = structEv->arrQT[no].Q;
95 eventOut[i].time = structEv->arrQT[no].time;
96 no--; i++;
97 if (no < 0) {
98 no = KOL_REMEMBER_QT_EVENT-1;
99 }
100 }
101
102 if (prTransition == true) {
103 structEv->pos = no;
104 }
105
106 return (i);
107}

150TEnumApiControlResult myFunctionControl (bool prChangeRamCoef)
151{
152 TApiControlEvent events[NUMBER_OF_EVENTS_STORED]; // Array with events
153 TEnumApiControlResult result = API_RESULT_OK;
154 uint8_t kolEvent;
155 TApiControlTargetValue target; //
156 float realTime; // real time
157 float realLeftW = 0, realRightW = 0;
158 float freqCall = 30;
159 static TControlEventQT evLeftPrev = {.Q = 0.f, .time = 0.f}, evRightPrev = {.Q = 0.f, .time = 0.f}; // value in beginning measure
160 static TControlEventQT evLeftCurr = {.Q = 0.f, .time = 0.f}, evRightCurr = {.Q = 0.f, .time = 0.f}; // current value
161 static TApiControlCoef currCoef = {0}; // Control regulator coefficients
162 static float outLeftS = 0, outRightS = 0; // Duty cycle
163 static TControlError errorLeft = {0}, errorRight = {0}; // PID error regulator
164 static TEventQT eventLeftQT = {0};
165 static TEventQT eventRightQT = {0};
166
167
168 // Get coefficients from RAM setting if need
169 if (prChangeRamCoef == true) {
170 apiTacControlFunc->GetCoefficients(&currCoef);
171 memset(&evLeftPrev, 0, sizeof(evLeftPrev));
172 memset(&evRightPrev, 0, sizeof(evRightPrev));
173 memset(&evLeftCurr, 0, sizeof(evLeftCurr));
174 memset(&evRightCurr, 0, sizeof(evRightCurr));
175 memset(&errorLeft, 0, sizeof(errorLeft));
176 memset(&errorRight, 0, sizeof(errorRight));
177 memset(&eventLeftQT, 0, sizeof(eventLeftQT));
178 memset(&eventRightQT, 0, sizeof(eventRightQT));
179 }
180
181 // Reading data
182 realTime = SCALE_TIME_API * (float)apiTacControlFunc->GetSystemTime_10Mcs();
183 kolEvent = apiTacControlFunc->GetLastEvents (events, NUMBER_OF_EVENTS_STORED);
184 freqCall = (float)apiTacControlFunc->GetFrequencyFunctionCall();
185
186
187 /***/
188 static float prevTime = 0.f;
189 static uint32_t ii = 0;
190 const float PING_TIME_INTERVAL = 5.f;
191 if (realTime - prevTime > PING_TIME_INTERVAL) {
192 prevTime = realTime;
193 apiTacControlFunc->ApiPrintf("Hello");
194 ii++;
195 }
196 /***/
197
198 // Remember last event and average W and T
205 for (uint8_t i = 0; i < kolEvent; i++) {
206 if (events[i].wheel == LEFT_WHEEL_API_CONTROL){
207 evLeftCurr.Q = events[i].angular;
208 evLeftCurr.time = SCALE_TIME_API * (float)events[i].time;
209 pushEventQT(evLeftCurr.Q, evLeftCurr.time, &eventLeftQT);
210 evLeftPrev = evLeftCurr;
211 }
212 if (events[i].wheel == RIGHT_WHEEL_API_CONTROL){
213 evRightCurr.Q = events[i].angular;
214 evRightCurr.time = SCALE_TIME_API * (float)events[i].time;
215 pushEventQT(evRightCurr.Q, evRightCurr.time, &eventRightQT);
216 evRightPrev = evRightCurr;
217 }
218 }
219
220
221 // CALCULATION ALL TELEMETRY
222 // Calculation angular velocity
223 uint8_t i = 0;
224 TControlEventQT lastEventLeft[KOL_REMEMBER_QT_EVENT] = {0}, lastEventRight[KOL_REMEMBER_QT_EVENT] = {0};
225 float dt = 0.f;
226 popEventQT(lastEventLeft, KOL_REMEMBER_QT_EVENT, &eventLeftQT, false);
227 popEventQT(lastEventRight, KOL_REMEMBER_QT_EVENT, &eventRightQT, false);
228 // Left wheel
229 dt = lastEventLeft[0].time-lastEventLeft[1].time;
230 while ((i < KOL_REMEMBER_QT_EVENT-2) && (dt < TIME_TO_CALC_W))
231 {
232 dt = lastEventLeft[0].time-lastEventLeft[i+2].time;
233 i++;
234 }
235 if (i == 0) {
236 realLeftW = 0.f;
237 } else {
238 if (dt > TIME_TO_CALC_W) {
239 dt = lastEventLeft[0].time - lastEventLeft[i].time;
240 realLeftW = (lastEventLeft[0].Q - lastEventLeft[i].Q) / dt;
241 } else {
242 realLeftW = (lastEventLeft[0].Q - lastEventLeft[i+1].Q) / dt;
243 }
244 }
245 if ((realTime - lastEventLeft[0].time) > TIME_TO_W_0) {
246 realLeftW = 0.f;
247 }
248 // Right wheel
249 i = 0; dt = 0.f;
250 dt = lastEventRight[0].time-lastEventRight[1].time;
251 while ((i < KOL_REMEMBER_QT_EVENT-2) && (dt < TIME_TO_CALC_W))
252 {
253 dt = lastEventRight[0].time-lastEventRight[i+2].time;
254 i++;
255 }
256 if (i == 0) {
257 realRightW = 0.f;
258 } else {
259 if (dt > TIME_TO_CALC_W) {
260 dt = lastEventRight[0].time - lastEventRight[i].time;
261 realRightW = (lastEventRight[0].Q - lastEventRight[i].Q) / dt;
262 } else {
263 realRightW = (lastEventRight[0].Q - lastEventRight[i+1].Q) / dt;
264 };
265 }
266 if ((realTime - lastEventRight[0].time) > TIME_TO_W_0) {
267 realRightW = 0.f;
268 }
269
270 // Calculation real Q of wheels
271 float realLeftQ, realRightQ;
272 // Left
273 float dQ = realLeftW*(realTime-lastEventLeft[0].time);
274 if (fabs(dQ) < SENSITIVITY_Q) {
275 realLeftQ = lastEventLeft[0].Q + dQ;
276 } else {
277 realLeftQ = lastEventLeft[0].Q + ((dQ > 0) ? SENSITIVITY_Q:(-SENSITIVITY_Q));
278 }
279 // Right
280 dQ = realRightW*(realTime-lastEventRight[0].time);
281 if (fabs(dQ) < SENSITIVITY_Q) {
282 realRightQ = lastEventRight[0].Q + dQ;
283 } else {
284 realRightQ = lastEventRight[0].Q + ((dQ > 0) ? SENSITIVITY_Q:(-SENSITIVITY_Q));
285 }
286
287 // CALCULATION OF CONTROL PARAMETERS
288 // Calculation duty cycle
289 target = apiTacControlFunc->GetTargetValue(realTime);
290 // Left wheel
291 if (target.left == TARGET_Q) {
292 errorLeft.eCurr = target.valueLeft - realLeftQ;
293 errorLeft.eIntegral = errorLeft.eIntegral + errorLeft.eCurr;
294 outLeftS = errorLeft.eCurr*currCoef.pidLeft.kProportional + errorLeft.eIntegral*currCoef.pidLeft.kIntegral/freqCall +
295 (errorLeft.eCurr - errorLeft.ePrev)*currCoef.pidLeft.kDifferential*freqCall;
296 }
297 if (target.left == TARGET_W) {
298 errorLeft.eCurr = target.valueLeft - realLeftW;
299 errorLeft.eIntegral = errorLeft.eIntegral + errorLeft.eCurr;
300 outLeftS = errorLeft.eCurr*currCoef.pidLeft.kProportional + errorLeft.eIntegral*currCoef.pidLeft.kIntegral/freqCall +
301 (errorLeft.eCurr - errorLeft.ePrev)*currCoef.pidLeft.kDifferential*freqCall;
302 }
303 errorLeft.ePrev = errorLeft.eCurr;
304 // Right wheel
305 if (target.right == TARGET_Q) {
306 errorRight.eCurr = target.valueRight - realRightQ;
307 errorRight.eIntegral = errorRight.eIntegral + errorRight.eCurr;
308 outRightS = errorRight.eCurr*currCoef.pidRight.kProportional + errorRight.eIntegral*currCoef.pidRight.kIntegral/freqCall +
309 (errorRight.eCurr - errorRight.ePrev)*currCoef.pidRight.kDifferential*freqCall;
310 }
311 if (target.right == TARGET_W) {
312 errorRight.eCurr = target.valueRight - realRightW;
313 errorRight.eIntegral = errorRight.eIntegral + errorRight.eCurr;
314 outRightS = errorRight.eCurr*currCoef.pidRight.kProportional + errorRight.eIntegral*currCoef.pidRight.kIntegral/freqCall +
315 (errorRight.eCurr - errorRight.ePrev)*currCoef.pidLeft.kDifferential*freqCall;
316 }
317 errorRight.ePrev = errorRight.eCurr;
318
319 // Send the calculated data
320 if (outLeftS > (100.f-EPSILON)){outLeftS = (100.f-EPSILON);}
321 if (outLeftS < (-100.f+EPSILON)){outLeftS = (-100.f+EPSILON);}
322 if (outRightS > (100.f-EPSILON)){outRightS = (100.f-EPSILON);}
323 if (outRightS < (-100.f+EPSILON)){outRightS = (-100.f+EPSILON);}
324 apiTacControlFunc->SetCoefficients(&currCoef);
325 apiTacControlFunc->SetVelocityAngular(realLeftW, realRightW);
326 apiTacControlFunc->SetDutyCycle(outLeftS, outRightS);
327
328 return (result);
329}
